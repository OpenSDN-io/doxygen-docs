52 XmppClient::XmppClient(EventManager *evm, const XmppChannelConfig *config)
53  : XmppConnectionManager(
54  evm, ssl::context::sslv23_client, config->auth_enabled, true),
55  config_mgr_(new XmppConfigManager),
56  lifetime_manager_(new LifetimeManager(
57  TaskScheduler::GetInstance()->GetTaskId("bgp::Config"))),
58  deleter_(new DeleteActor(this)),
59  auth_enabled_(config->auth_enabled),
60  tcp_hold_time_(config->tcp_hold_time) {
61 
62  if (config->auth_enabled) {
63 
64  // Get SSL context from base class and update
65  boost::asio::ssl::context *ctx = context();
66  boost::system::error_code ec;
67 
68  //set mode
69  ctx->set_options(ssl::context::default_workarounds |
70  ssl::context::no_sslv3 | ssl::context::no_sslv2 | ssl::context::no_tlsv1, ec);
71  if (ec.value() != 0) {
72  LOG(ERROR, "Error : " << ec.message() << ", setting ssl options");
73  exit(EINVAL);
74  }
75 
76  // CA certificate, used to verify if the peer certificate
77  // is signed by a trusted CA
78  std::string ca_cert_filename = config->path_to_ca_cert;
79  if (!ca_cert_filename.empty()) {
80 
81  // Verify peer has CA signed certificate
82  ctx->set_verify_mode(boost::asio::ssl::verify_peer, ec);
83  if (ec.value() != 0) {
84  LOG(ERROR, "Error : " << ec.message()
85  << ", while setting ssl verification mode");
86  exit(EINVAL);
87  }
88 
89  ctx->load_verify_file(config->path_to_ca_cert, ec);
90  if (ec.value() != 0) {
91  LOG(ERROR, "Error : " << ec.message()
92  << ", while using cacert file : "
93  << config->path_to_ca_cert);
94  exit(EINVAL);
95  }
96  }
97 
98  // server certificate
99  ctx->use_certificate_chain_file(config->path_to_server_cert, ec);
100  if (ec.value() != 0) {
101  LOG(ERROR, "Error : " << ec.message() <<
102  ", while using server cert file : "
103  << config->path_to_server_cert);
104  exit(EINVAL);
105  }
106 
107  // server private key
108  ctx->use_private_key_file(config->path_to_server_priv_key,
109  boost::asio::ssl::context::pem, ec);
110  if (ec.value() != 0) {
111  LOG(ERROR, "Error : " << ec.message()
112  << ", while using privkey file : "
113  << config->path_to_server_priv_key);
114  exit(EINVAL);
115  }
116  }

135 TcpSession *XmppClient::CreateSession() {
136  typedef boost::asio::detail::socket_option::boolean<
137 #ifdef __APPLE__
138  SOL_SOCKET, SO_REUSEPORT> reuse_port_t;
139 #else
140  SOL_SOCKET, SO_REUSEADDR> reuse_addr_t;
141 #endif
142 
143  TcpSession *session = TcpServer::CreateSession();
144  Socket *socket = session->socket();
145 
146  boost::system::error_code err;
147  socket->open(ip::tcp::v4(), err);
148  if (err) {
149  XMPP_WARNING(ClientOpenFail, session->ToUVEKey(), XMPP_PEER_DIR_OUT,
150  err.message());
151  DeleteSession(session);
152  return NULL;
153  }
154 
155 #ifdef __APPLE__
156  socket->set_option(reuse_port_t(true), err);
157 #else
158  socket->set_option(reuse_addr_t(true), err);
159 #endif
160  if (err) {
161  XMPP_WARNING(SetSockOptFail, session->ToUVEKey(), XMPP_PEER_DIR_OUT,
162  err.message());
163  return session;
164  }
165 
166  err = session->SetSocketOptions();
167  if (err) {
168  DeleteSession(session);
169  assert(0);
170  }
171 
172  return session;
173 }

181 XmppClient::ProcessConfigUpdate(XmppConfigManager::DiffType delta,
182  const XmppChannelConfig *current, const XmppChannelConfig *future) {
183  if (delta == XmppConfigManager::DF_ADD) {
184  XmppClientConnection *connection = CreateConnection(future);
185  connection->Initialize(); // trigger state-machine
186  }
187  if (delta == XmppConfigManager::DF_DELETE) {
188  ConnectionMap::iterator loc = connection_map_.find(current->endpoint);
189  if (loc != connection_map_.end()) {
190  loc->second->ManagedDelete();
191  }
192  }
193 }
