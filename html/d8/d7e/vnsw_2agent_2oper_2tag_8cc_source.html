70 bool TagEntry::Change(TagTable *table, const DBRequest *req) {
71  bool ret = false;
72  bool insert = false;
73  TagData *data = static_cast<TagData *>(req->data.get());
74  uint32_t old_tag_id;
75 
76  if (tag_id_ != data->tag_id_) {
77  old_tag_id = tag_id_;
78  tag_id_ = data->tag_id_;
79  insert = true;
80  ret = true;
81  }
82 
83  //Typically tag of type application would have one link to
84  //application policy set, but there is no commit check to verify
85  //that there is only one link.
86  //In case there are multiple links to application policy set
87  //precedence would be given to APS that is defined at global level
88  //Orden among global level application policy set is not defined
89  AgentDBTable *ps_table = table->agent()->policy_set_table();
90  PolicySetList new_policy_list;
91  TagData::PolicySetUuidList::const_iterator it =
92  data->policy_set_uuid_list_.begin();
93  for(; it != data->policy_set_uuid_list_.end(); it++) {
94  PolicySetKey key(*it);
95  PolicySet *aps =
96  static_cast<PolicySet *>(ps_table->FindActiveEntry(&key));
97  if (aps) {
98  new_policy_list.push_back(aps);
99  }
100  }
101 
102  if (policy_set_list_ != new_policy_list) {
103  policy_set_list_ = new_policy_list;
104  ret = true;
105  }
106 
107  if (tag_type_ != data->tag_type_) {
108  tag_type_ = data->tag_type_;
109  ret = true;
110  }
111 
112  if (tag_value_ != data->tag_value_) {
113  tag_value_ = data->tag_value_;
114  ret = true;
115  }
116 
117  if (insert) {
118  table->Erase(old_tag_id);
119  table->Insert(tag_id_, name_);
120  }
121 
122  return ret;
123 }

232 TagData* TagTable::BuildData(Agent *agent, IFMapNode *node) {
233  Tag *cfg = static_cast<Tag *>(node->GetObject());
234  //Parse thru all the application policy set attached to a tag
235  //If there are mutliple application policy set attached to a tag
236  //First priority would be given to APS defined at global level
237  //and the APS defined at project level
238  TagData::PolicySetUuidList prj_policy_set;
239  TagData::PolicySetUuidList global_policy_set;
240 
241  IFMapAgentTable *table = static_cast<IFMapAgentTable *>(node->table());
242  for (DBGraphVertex::adjacency_iterator iter =
243  node->begin(table->GetGraph());
244  iter != node->end(table->GetGraph()); ++iter) {
245 
246  boost::uuids::uuid aps_uuid(boost::uuids::nil_uuid());
247  IFMapNode *adj_node = static_cast<IFMapNode *>(iter.operator->());
248  if (agent->config_manager()->SkipNode(adj_node,
249  agent->cfg()->cfg_policy_set_table())) {
250  continue;
251  }
252 
253  ApplicationPolicySet *aps =
254  static_cast<ApplicationPolicySet *>(adj_node->GetObject());
255  autogen::IdPermsType id_perms = aps->id_perms();
256  CfgUuidSet(id_perms.uuid.uuid_mslong, id_perms.uuid.uuid_lslong,
257  aps_uuid);
258 
259  if (IsGlobalAps(agent, adj_node)) {
260  global_policy_set.push_back(aps_uuid);
261  } else {
262  prj_policy_set.push_back(aps_uuid);
263  }
264  }
265 
266  TagData *data = new TagData(agent, node,
267  strtol(cfg->id().c_str(), NULL, 16));
268  data->policy_set_uuid_list_ = global_policy_set;
269  data->policy_set_uuid_list_.insert(data->policy_set_uuid_list_.end(),
270  prj_policy_set.begin(),
271  prj_policy_set.end());
272  data->name_ = node->name();
273  data->tag_type_ = TagEntry::GetTypeVal(cfg->type_name(), cfg->id());
274  data->tag_value_ = cfg->value();
275  return data;
276 }
