137  bool AddIpam(const Ip4Address &ip, uint8_t plen) {
138  bool ret = false;
139  IpSubnet ips(ip, plen);
140  if (ipam_subnet_.find(ips) == ipam_subnet_.end()) {
141  ipam_subnet_[ips] = 0;
142  ret = true;
143  }
144  ipam_subnet_[ips]++;
145  return ret;
146  }
147 
148  bool DelIpam(const Ip4Address &ip, uint8_t plen) {
149  IpSubnet ips(ip, plen);
150  if (ipam_subnet_.find(ips) == ipam_subnet_.end()) {
151  return false;
152  }
153 
154  ipam_subnet_[ips]--;
155  if (ipam_subnet_[ips] == 0) {
156  ipam_subnet_.erase(ips);
157  return true;
158  }
159 
160  return false;
161  }
