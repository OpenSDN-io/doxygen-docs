139  bool AddIpam(const Ip4Address &ip, uint8_t plen) {
140  bool ret = false;
141  IpSubnet ips(ip, plen);
142  if (ipam_subnet_.find(ips) == ipam_subnet_.end()) {
143  ipam_subnet_[ips] = 0;
144  ret = true;
145  }
146  ipam_subnet_[ips]++;
147  return ret;
148  }
149 
150  bool DelIpam(const Ip4Address &ip, uint8_t plen) {
151  IpSubnet ips(ip, plen);
152  if (ipam_subnet_.find(ips) == ipam_subnet_.end()) {
153  return false;
154  }
155 
156  ipam_subnet_[ips]--;
157  if (ipam_subnet_[ips] == 0) {
158  ipam_subnet_.erase(ips);
159  return true;
160  }
161 
162  return false;
163  }
