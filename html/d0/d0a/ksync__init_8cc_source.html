260 void KSync::SetHugePages() {
261  vr_hugepage_config encoder;
262  bool fail[kHugePageFiles];
263  std::string filename[kHugePageFiles];
264  uint32_t filesize[kHugePageFiles];
265  uint32_t flags[kHugePageFiles];
266  uint32_t pagesize[kHugePageFiles];
267  uint16_t i, j;
268  uint32_t bridge_table_size, flow_table_size;
269 
270  // get the table size for bridge and flow
271  bridge_table_size = ksync_bridge_memory_.get()->GetKernelTableSize();
272  flow_table_size = ksync_flow_memory_.get()->GetKernelTableSize();
273 
274  LOG(INFO, __FUNCTION__ << ": " << "Bridge table size:" << bridge_table_size
275  << " Flow table size:" << flow_table_size << "\n");
276 
277  for (i = 0; i < kHugePageFiles / 2; ++i) {
278  filename[i] = agent_->params()->huge_page_file_1G(i);
279  if ((i % 2) == 0) {
280  filesize[i] = bridge_table_size;
281  } else {
282  filesize[i] = flow_table_size;
283  }
284  // set pagesize array
285  pagesize[i] = 1024 * 1024 * 1024;
286  flags[i] = O_CREAT | O_RDWR;
287  fail[i] = false;
288  }
289  for (j = i; j < kHugePageFiles; ++j) {
290  filename[j] = agent_->params()->huge_page_file_2M(j - i);
291  if ((j % 2) == 0) {
292  filesize[j] = bridge_table_size;
293  } else {
294  filesize[j] = flow_table_size;
295  }
296  // set pagesize array
297  pagesize[j] = 2 * 1024 * 1024;
298  flags[j] = O_CREAT | O_RDWR;
299  fail[j] = false;
300  }
301 
302  for (i = 0; i < kHugePageFiles; ++i) {
303  if (filename[i].empty()) {
304  fail[i] = true;
305  continue;
306  }
307 
308  huge_fd_[i] = open(filename[i].c_str(), flags[i], 0755);
309  if (huge_fd_[i] < 0) {
310  fail[i] = true;
311  continue;
312  }
313 
314  LOG(INFO, "Mem mapping hugepage file:" << filename[i].c_str()
315  << " size:" << filesize[i] << "\n");
316  huge_pages_[i] = (void *) mmap(NULL, filesize[i],
317  PROT_READ | PROT_WRITE, MAP_SHARED,
318  huge_fd_[i], 0);
319  if (huge_pages_[i] == MAP_FAILED) {
320  LOG(ERROR, "Failed to Mmap hugepage file:" << filename[i].c_str() << "\n");
321  huge_pages_[i] = NULL;
322  fail[i] = true;
323  } else {
324  LOG(INFO, "Mem mapped hugepage file:" << filename[i].c_str()
325  << " to addr:" << huge_pages_[i] << "\n");
326  if ((i % 2) == 0) {
327  btable_huge_pages_index_ = i;
328  } else {
329  ftable_huge_pages_index_ = i;
330  }
331  }
332  }
333 
334  encoder.set_vhp_op(sandesh_op::ADD);
335 
336  std::vector<uint64_t> huge_mem;
337  std::vector<uint32_t> huge_mem_size;
338  std::vector<uint32_t> huge_page_size;
339  std::vector<int8_t> huge_page_paths;
340  std::vector<uint32_t> huge_page_paths_sz;
341 
342  for (uint16_t i = 0; i < kHugePageFiles; ++i) {
343  if (fail[i] == false) {
344  huge_mem.push_back((uint64_t) huge_pages_[i]);
345  huge_page_size.push_back(pagesize[i]);
346  huge_mem_size.push_back(filesize[i]);
347  const char *path = filename[i].c_str();
348  uint32_t len = strlen(path) + 1;
349  for (uint32_t c = 0; c < len; c++) {
350  huge_page_paths.push_back(path[c]);
351  }
352  huge_page_paths_sz.push_back(len);
353  }
354  }
355  encoder.set_vhp_mem(huge_mem);
356  encoder.set_vhp_psize(huge_page_size);
357  // set huge_mem_size
358  encoder.set_vhp_mem_sz(huge_mem_size);
359  encoder.set_vhp_file_paths(huge_page_paths);
360  encoder.set_vhp_file_path_sz(huge_page_paths_sz);
361  encoder.set_vhp_resp(VR_HPAGE_CFG_RESP_HPAGE_SUCCESS);
362 
363  uint8_t msg[KSYNC_DEFAULT_MSG_SIZE];
364  int len = Encode(encoder, msg, KSYNC_DEFAULT_MSG_SIZE);
365 
366  LOG(INFO, "Sending Huge Page configuration to VROUTER\n");
367  KSyncSock *sock = KSyncSock::Get(0);
368  sock->BlockingSend((char *)msg, len);
369  if (sock->BlockingRecv()) {
370  LOG(ERROR, "Error sending Huge Page configuration to VROUTER. Skipping KSync Start");
371  }
372 }

374 void KSync::ResetVRouter(bool run_sync_mode) {
375  int len = 0;
376  vrouter_ops encoder;
377  encoder.set_h_op(sandesh_op::RESET);
378  uint8_t msg[KSYNC_DEFAULT_MSG_SIZE];
379  len = Encode(encoder, msg, KSYNC_DEFAULT_MSG_SIZE);
380 
381  KSyncSock *sock = KSyncSock::Get(0);
382  sock->BlockingSend((char *)msg, len);
383  if (sock->BlockingRecv()) {
384  LOG(ERROR, "Error resetting VROUTER. Skipping KSync Start");
385  return;
386  }
387 
388  //configure vrouter with priority_tagging configuration
389  encoder.set_h_op(sandesh_op::ADD);
390  encoder.set_vo_priority_tagging(agent_->params()->qos_priority_tagging());
391  //Initialize rest of the fields to values so that vrouter does not take any
392  //action on those field values
393  InitVrouterOps(&encoder);
394  len = Encode(encoder, msg, KSYNC_DEFAULT_MSG_SIZE);
395  sock->BlockingSend((char *)msg, len);
396  if (sock->BlockingRecv()) {
397  LOG(ERROR, "Error setting Qos priority-tagging for vrouter");
398  }
399 
400  //Get configured mpls, vmi, vni and nexthop parameters
401  //from vrouter
402  encoder.set_h_op(sandesh_op::GET);
403  len = Encode(encoder, msg, KSYNC_DEFAULT_MSG_SIZE);
404  sock->BlockingSend((char *)msg, len);
405  if (sock->BlockingRecv()) {
406  LOG(ERROR, "Error getting configured parameter for vrouter");
407  }
408 
409  KSyncSock::Start(run_sync_mode);
410 }

416 void KSync::CreateVhostIntf() {

424  ret = nl_connect(cl, 0, 0, 0);
425  if(ret != 0) {
426  if(!(cl->cl_sa))
427  LogSockInitErrors(false, false, "cl_sa malloc failure");
428  else
429  LogSockInitErrors(false, true, "connect");
430  }
431 
432  struct vn_if ifm;
433  struct nl_response *resp;
434 
435  memset(&ifm, 0, sizeof(ifm));
436  strncpy(ifm.if_name, agent_->vhost_interface_name().c_str(),
437  IFNAMSIZ);
438  ifm.if_name[IFNAMSIZ - 1] = '\0';
439  strcpy(ifm.if_kind, VHOST_KIND);
440  ifm.if_flags = IFF_UP;
441 
442  assert(nl_build_if_create_msg(cl, &ifm, 1) == 0);

444  ret = nl_recvmsg(cl);

446  LogSockInitErrors(false, false, "cl->cl_recv_len > cl->cl_buf_len");
447  } else if (ret < 0) {
448  LogSockInitErrors(false, true, "recvmsg");
449  }
450  assert((resp = nl_parse_reply(cl)) != NULL);
451  assert(resp->nl_type == NL_MSG_TYPE_ERROR);
452  nl_free_client(cl);
453 #elif defined(__FreeBSD__)
454  struct ifreq ifr;
455  memset(&ifr, 0, sizeof(ifr));
456  ifr.ifr_flags = IFF_UP;
