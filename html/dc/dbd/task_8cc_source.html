257 // Method called from tbb::task to execute.
258 // Invoke Run() method of client.
259 // Supports task continuation when Run() returns false
260 tbb::task *TaskImpl::execute() {
261  TaskInfo::reference running = task_running.local();
262  running = parent_;
263  parent_->SetTbbState(Task::TBB_EXEC);
264  try {
265  uint64_t t = 0;
266  if (parent_->enqueue_time() != 0) {
267  t = ClockMonotonicUsec();
268  TaskScheduler *scheduler = TaskScheduler::GetInstance();
269  if ((t - parent_->enqueue_time()) >
270  scheduler->schedule_delay(parent_)) {
271  TASK_TRACE(scheduler, parent_, "TBB schedule time(in usec) ",
272  (t - parent_->enqueue_time()));
273  }
274  } else if (TaskScheduler::GetInstance()->track_run_time()) {
275  t = ClockMonotonicUsec();
276  }
277 
278  bool is_complete = parent_->Run();
279  if (t != 0) {
280  int64_t delay = ClockMonotonicUsec() - t;
281  TaskScheduler *scheduler = TaskScheduler::GetInstance();
282  uint32_t execute_delay = scheduler->execute_delay(parent_);
283  if (execute_delay && delay > execute_delay) {
284  TASK_TRACE(scheduler, parent_, "Run time(in usec) ", delay);
285  }
286  if (scheduler->track_run_time()) {
287  TaskGroup *group =
288  scheduler->QueryTaskGroup(parent_->GetTaskId());
289  group->IncrementTotalRunTime(delay);
290  }
291  }
292 
293  running = NULL;
294  if (is_complete == true) {
295  parent_->SetTaskComplete();
296  } else {
297  parent_->SetTaskRecycle();
298  }
299  } catch (std::exception &e) {
300 
301  // Store exception information statically, to easily read exception
302  // information from the core.
303  static std::string what = e.what();
304 
305  LOG(ERROR, "!!!! ERROR !!!! Task caught fatal exception: " << what
306  << " TaskImpl: " << this);
307  assert(0);
308  } catch (...) {
309  LOG(ERROR, "!!!! ERROR !!!! Task caught fatal unknown exception"
310  << " TaskImpl: " << this);
311  assert(0);
312  }
313 
314  return NULL;
315 }

587 void TaskScheduler::EnqueueUnLocked(Task *t) {
588  if (measure_delay_) {
589  t->enqueue_time_ = ClockMonotonicUsec();
590  }
591  // Ensure that task is enqueued only once.
592  assert(t->GetSeqno() == 0);
593  enqueue_count_++;
594  t->SetSeqNo(++seqno_);
595  TaskGroup *group = GetTaskGroup(t->GetTaskId());
596  t->schedule_delay_ = group->schedule_delay_;
597  t->execute_delay_ = group->execute_delay_;
598  group->stats_.enqueue_count_++;
599 
600  TaskEntry *entry = GetTaskEntry(t->GetTaskId(), t->GetTaskInstance());
601  entry->stats_.enqueue_count_++;
602  // If either TaskGroup or TaskEntry is disabled for Unit-Test purposes,
603  // enqueue new task in waitq and update TaskGroup if needed.
604  if (group->IsDisabled() || entry->IsDisabled()) {
605  entry->AddToWaitQ(t);
606  if (group->IsDisabled()) {
607  group->AddToDisableQ(entry);
608  }
609  return;
610  }
611 
612  // Add task to waitq_ if its already populated
613  if (entry->WaitQSize() != 0) {
614  entry->AddToWaitQ(t);
615  return;
616  }
617 
618  // Is scheduler stopped? Dont add task to deferq_ if scheduler is stopped.
619  // TaskScheduler::Start() will run tasks from waitq_
620  if (running_ == false) {
621  entry->AddToWaitQ(t);
622  stop_entry_->AddToDeferQ(entry);
623  return;
624  }
625 
626  // Check Task Group policy. On policy violation, DeferOnPolicyFail()
627  // adds the Task to the TaskEntry's waitq_ and the TaskEntry will be
628  // added to deferq_ of the matching TaskGroup.
629  if (group->DeferOnPolicyFail(entry, t)) {
630  return;
631  }
632 
633  // Check Task Entry policy. On policy violation, DeferOnPolicyFail()
634  // adds the Task to the TaskEntry's waitq_ and the TaskEntry will be
635  // added to deferq_ of the matching TaskEntry.
636  if (entry->DeferOnPolicyFail(t)) {
637  return;
638  }
639 
640  entry->RunTask(t);
641  return;
642 }

649  // If the task is in RUN state, mark the task for cancellation and return.
650  if (t->state_ == Task::RUN) {
651  t->task_cancel_ = true;
652  } else if (t->state_ == Task::WAIT) {
653  TaskEntry *entry = QueryTaskEntry(t->GetTaskId(), t->GetTaskInstance());
654  TaskGroup *group = QueryTaskGroup(t->GetTaskId());
655  assert(entry->WaitQSize());
656  // Get the first entry in the waitq_
657  Task *first_wait_task = &(*entry->waitq_.begin());
658  TaskEntry *disable_entry = group->GetDisableEntry();
659  assert(entry->DeleteFromWaitQ(t) == true);
660  // If the waitq_ is empty, then remove the TaskEntry from the deferq.
661  if (!entry->WaitQSize()) {
662  if (entry->deferq_task_group_) {
663  assert(entry->deferq_task_entry_ == NULL);
664  entry->deferq_task_group_->DeleteFromDeferQ(*entry);
665  } else if (entry->deferq_task_entry_) {
666  entry->deferq_task_entry_->DeleteFromDeferQ(*entry);
667  } else if (group->IsDisabled()) {
668  // Remove TaskEntry from deferq of disable_entry
669  disable_entry->DeleteFromDeferQ(*entry);
670  } else {
671  if (!entry->IsDisabled()) {
672  assert(0);
673  }
674  }
675  } else if (t == first_wait_task) {
676  // TaskEntry is inserted in the deferq_ based on the Task seqno.
677  // deferq_ comparison function uses the seqno of the first entry in
678  // the waitq_. Therefore, if the task to be cancelled is the first
679  // entry in the waitq_, then delete the entry from the deferq_ and
680  // add it again.
681  TaskGroup *deferq_tgroup = entry->deferq_task_group_;
682  TaskEntry *deferq_tentry = entry->deferq_task_entry_;
683  if (deferq_tgroup) {
684  assert(deferq_tentry == NULL);
685  deferq_tgroup->DeleteFromDeferQ(*entry);
686  deferq_tgroup->AddToDeferQ(entry);
687  } else if (deferq_tentry) {
688  deferq_tentry->DeleteFromDeferQ(*entry);
689  deferq_tentry->AddToDeferQ(entry);
690  } else if (group->IsDisabled()) {
691  // Remove TaskEntry from deferq of disable_entry and add back
692  disable_entry->DeleteFromDeferQ(*entry);
693  disable_entry->AddToDeferQ(entry);
694  } else {
695  if (!entry->IsDisabled()) {
696  assert(0);
697  }
698  }
699  }
700  delete t;
701  cancel_count_++;
702  return CANCELLED;
703  } else {
704  return FAILED;
705  }
706  return QUEUED;
707 }

876 //
877 // Platfrom-dependent subroutine in Linux and FreeBSD implementations,
878 // used only in TaskScheduler::WaitForTerminateCompletion()
879 //
880 // In Linux, make sure that all the [tbb] threads launched have completely
881 // exited. We do so by looking for the Threads count of this process in
882 // /proc/<pid>/status
883 //
884 // In FreeBSD use libprocstat to check how many threads is running
885 // in specific process.
886 //
887 int TaskScheduler::CountThreadsPerPid(pid_t pid) {
888  int threads;
889  threads = 0;
890 
891 #if defined(__FreeBSD__)
892  struct kinfo_proc *ki_proc;
893  struct procstat *pstat;
894  unsigned int count_procs;
895 
896 
897  count_procs = 0;
898 
899  pstat = procstat_open_sysctl();
900  if(pstat == NULL) {
901  LOG(ERROR, "procstat_open_sysctl() failed");
902  return -1;
903  }
904 
905  ki_proc = procstat_getprocs(pstat, KERN_PROC_PID, pid, &count_procs);
906  if (ki_proc == NULL) {
907  LOG(ERROR, "procstat_open_sysctl() failed");
908  return -1;
909  }
910 
911  if (count_procs != 0)
912  procstat_getprocs(pstat, KERN_PROC_PID | KERN_PROC_INC_THREAD,
913  ki_proc->ki_pid, &threads);
914 
915  procstat_freeprocs(pstat, ki_proc);
916  procstat_close(pstat);
917 
918 #elif defined(__linux__)
919  std::ostringstream file_name;
920  std::string line;
921 
922  file_name << "/proc/" << pid << "/status";
923 
924  std::ifstream file(file_name.str().c_str());
925 
926  if(!file) {
927  LOG(ERROR, "opening /proc failed");
928  return -1;
929  }
930 
931  while (threads == 0 && file.good()) {
932  getline(file, line);
933  if (line == "Threads:\t1") threads = 1;
934  }
935  file.close();
936 #else
937 #error "TaskScheduler::CountThreadsPerPid() - unsupported platform."
938 #endif
939 
940  return threads;
941 }
