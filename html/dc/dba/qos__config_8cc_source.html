211 bool AgentQosConfig::Change(const DBRequest *req) {
212  bool ret = false;
213  const AgentQosConfigData *data =
214  static_cast<const AgentQosConfigData *>(req->data.get());
215 
216  //Ignore change notification if entry node is marked for deletion
217  if ((data->ifmap_node() == NULL) || (data->ifmap_node()->IsDeleted())) {
218  return ret;
219  }
220 
221  if (name_ != data->name_) {
222  name_ = data->name_;
223  ret = true;
224  }
225 
226  if (HandleQosForwardingMapChange(data->agent(), dscp_map_, data->dscp_map_)) {
227  ret = true;
228  }
229 
230  if (HandleQosForwardingMapChange(data->agent(), vlan_priority_map_,
231  data->vlan_priority_map_)) {
232  ret = true;
233  }
234 
235  if (HandleQosForwardingMapChange(data->agent(), mpls_exp_map_,
236  data->mpls_exp_map_)) {
237  ret = true;
238  }
239 
240  if (type_ != data->type_) {
241  type_ = data->type_;
242  ret = true;
243  }
244 
245  if (type_ != DEFAULT) {
246  HandleGlobalQosConfig(data);
247  }
248 
249  if (default_forwarding_class_ != data->default_forwarding_class_) {
250  default_forwarding_class_ = data->default_forwarding_class_;
251  ret = true;
252  }
253 
254  return ret;
255 }
