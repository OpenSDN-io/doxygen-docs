21 size_t EvpnTable::HashFunction(const EvpnPrefix &prefix) {
22  if (prefix.type() == EvpnPrefix::MacAdvertisementRoute) {
23  if (prefix.mac_addr().IsBroadcast())
24  return 0;
25  const uint8_t *data = prefix.mac_addr().GetData();
26  uint32_t value = get_value(data + 2, 4);
27  return boost::hash_value(value);
28  }
29  if (prefix.type() == EvpnPrefix::IpPrefixRoute) {
30  if (prefix.ip_address().is_v4()) {
31  return InetTable::HashFunction(prefix.inet_prefix());
32  } else {
33  return Inet6Table::HashFunction(prefix.inet6_prefix());
34  }
35  }
36  return 0;
37 }

59 void EvpnTable::AddRemoveCallback(const DBEntryBase *entry, bool add) const {
60  if (IsVpnTable())
61  return;
62  const EvpnRoute *evpn_rt = static_cast<const EvpnRoute *>(entry);
63  const EvpnPrefix &evpn_prefix = evpn_rt->GetPrefix();
64  switch (evpn_prefix.type()) {
65  case EvpnPrefix::MacAdvertisementRoute:
66  // Ignore Broadcast MAC routes.
67  if (evpn_prefix.mac_addr().IsBroadcast())
68  break;
69 
70  if (add) {
71  mac_route_count_++;
72  } else {
73  mac_route_count_--;
74  }
75 
76  // Ignore MAC routes with IP addresses.
77  if (evpn_prefix.family() != Address::UNSPEC)
78  break;
79 
80  if (add) {
81  unique_mac_route_count_++;
82  } else {
83  unique_mac_route_count_--;
84  }
85  break;
86 
87  case EvpnPrefix::InclusiveMulticastRoute:
88  if (add) {
89  im_route_count_++;
90  } else {
91  im_route_count_--;
92  }
93  break;
94 
95  case EvpnPrefix::IpPrefixRoute:
96  if (add) {
97  ip_route_count_++;
98  } else {
99  ip_route_count_--;
100  }
101  break;
102 
103  default:
104  break;
105  }
106 }

154 bool EvpnTable::ShouldReplicate(const BgpServer *server,
155  const BgpTable *src_table,
156  const ExtCommunityPtr community,
157  const EvpnPrefix &evpn_prefix) const {
158  // Always replicate into master table.
159  if (IsMaster())
160  return true;
161 
162  // Always replicate Type-5 routes.
163  if (evpn_prefix.type() == EvpnPrefix::IpPrefixRoute)
164  return true;
165 
166  // Don't replicate to a VRF from other VRF tables.
167  const EvpnTable *src_evpn_table =
168  dynamic_cast<const EvpnTable *>(src_table);
169  if (!src_evpn_table->IsMaster())
170  return false;
171 
172  // Replicate to VRF from the VPN table if OriginVn matches.
173  if (community->ContainsOriginVn(server->autonomous_system(),
174  routing_instance()->virtual_network_index())) {
175  return true;
176  }
177 
178  // Do not replicate non AD routes as the OriginVN does not match.
179  if (evpn_prefix.type() != EvpnPrefix::AutoDiscoveryRoute)
180  return false;
181 
182  string es_target = server->autonomous_system() > 0xffFF ?
183  integerToString(EVPN_ES_IMPORT_ROUTE_TARGET_AS4) :
184  integerToString(EVPN_ES_IMPORT_ROUTE_TARGET_AS2);
185 
186  // Replicate if AD route target is associated with the route.
187  RouteTarget rtarget = RouteTarget::FromString("target:" +
188  integerToString(server->autonomous_system()) + ":" + es_target);
189  if (community->ContainsRTarget(rtarget.GetExtCommunity()))
190  return true;
191  return false;
192 }

194 BgpRoute *EvpnTable::RouteReplicate(BgpServer *server,
195  BgpTable *src_table, BgpRoute *src_rt, const BgpPath *src_path,
196  ExtCommunityPtr community) {
197  assert(src_table->family() == Address::EVPN);
198  EvpnRoute *evpn_rt = dynamic_cast<EvpnRoute *>(src_rt);
199  assert(evpn_rt);
200  EvpnPrefix evpn_prefix(evpn_rt->GetPrefix());
201 
202  // Check if this evpn route should be replicated.
203  if (!ShouldReplicate(server, src_table, community, evpn_prefix))
204  return NULL;
205  if (evpn_prefix.type() == EvpnPrefix::AutoDiscoveryRoute) {
206  if (IsMaster() || evpn_prefix.tag() != EvpnPrefix::kMaxTag)
207  return NULL;
208  community = server->extcomm_db()->ReplaceRTargetAndLocate(
209  community.get(), ExtCommunity::ExtCommunityList());
210  }
211  if (evpn_prefix.type() == EvpnPrefix::SegmentRoute)
212  return NULL;
213  if (evpn_prefix.type() == EvpnPrefix::MacAdvertisementRoute &&
214  evpn_prefix.mac_addr().IsBroadcast())
215  return NULL;
216 
217  BgpAttrDB *attr_db = server->attr_db();
218  BgpAttrPtr new_attr(src_path->GetAttr());
219 
220  if (IsMaster()) {
221  if (evpn_prefix.route_distinguisher().IsZero()) {
222  evpn_prefix.set_route_distinguisher(new_attr->source_rd());
223  }
224  } else {
225  if (evpn_prefix.type() == EvpnPrefix::AutoDiscoveryRoute ||
226  evpn_prefix.type() == EvpnPrefix::MacAdvertisementRoute ||
227  evpn_prefix.type() == EvpnPrefix::IpPrefixRoute) {
228  evpn_prefix.set_route_distinguisher(RouteDistinguisher::kZeroRd);
229  }
230  }
231  EvpnRoute rt_key(evpn_prefix);
232 
233  // Find or create the route.
234  DBTablePartition *rtp =
235  static_cast<DBTablePartition *>(GetTablePartition(&rt_key));
236  BgpRoute *dest_route = static_cast<BgpRoute *>(rtp->Find(&rt_key));
237  if (dest_route == NULL) {
238  dest_route = new EvpnRoute(evpn_prefix);
239  rtp->Add(dest_route);
240  } else {
241  dest_route->ClearDelete();
242  }
243 
244  new_attr = attr_db->ReplaceExtCommunityAndLocate(new_attr.get(), community);
245 
246  // Check whether peer already has a path
247  BgpPath *dest_path = dest_route->FindSecondaryPath(src_rt,
248  src_path->GetSource(), src_path->GetPeer(),
249  src_path->GetPathId());
250  if (dest_path != NULL) {
251  if ((new_attr != dest_path->GetOriginalAttr()) ||
252  (src_path->GetFlags() != dest_path->GetFlags()) ||
253  (src_path->GetLabel() != dest_path->GetLabel()) ||
254  (src_path->GetL3Label() != dest_path->GetL3Label())) {
255  if (dest_path->NeedsResolution()) {
256  path_resolver()->StopPathResolution(rtp->index(), dest_path);
257  }
258  bool success = dest_route->RemoveSecondaryPath(src_rt,
259  src_path->GetSource(), src_path->GetPeer(),
260  src_path->GetPathId());
261  assert(success);
262  } else {
263  return dest_route;
264  }
265  }
266 
267  // Create replicated path and insert it on the route
268  BgpSecondaryPath *replicated_path =
269  new BgpSecondaryPath(src_path->GetPeer(), src_path->GetPathId(),
270  src_path->GetSource(), new_attr,
271  src_path->GetFlags(), src_path->GetLabel(),
272  src_path->GetL3Label());
273  replicated_path->SetReplicateInfo(src_table, src_rt);
274 
275  // For VPN to VRF replication, start path resolution if fast convergence is
276  // enabled and update path flag to indicate need for resolution.
277  if (!IsMaster() && server->IsNextHopCheckEnabled() &&
278  (replicated_path->GetSource() == BgpPath::BGP_XMPP)) {
279  Address::Family family = src_path->GetAttr()->nexthop_family();
280  RoutingInstanceMgr *mgr = server->routing_instance_mgr();
281  RoutingInstance *master_ri = mgr->GetDefaultRoutingInstance();
282  BgpTable *table = master_ri->GetTable(family);
283  replicated_path->SetResolveNextHop();
284  path_resolver()->StartPathResolution(dest_route,
285  replicated_path, table);
286  }
287 
288  dest_route->InsertPath(replicated_path);
289 
290  // Always trigger notification.
291  rtp->Notify(dest_route);
292 
293  return dest_route;
294 }

296 bool EvpnTable::Export(RibOut *ribout, Route *route,
297  const RibPeerSet &peerset, UpdateInfoSList &uinfo_slist) {
298  EvpnRoute *evpn_route = dynamic_cast<EvpnRoute *>(route);
299  assert(evpn_route);
300 
301  if (ribout->IsEncodingBgp()) {
302  UpdateInfo *uinfo = GetUpdateInfo(ribout, evpn_route, peerset);
303  if (!uinfo)
304  return false;
305  uinfo_slist->push_front(*uinfo);
306  return true;
307  }
308 
309  const EvpnPrefix &evpn_prefix = evpn_route->GetPrefix();
310  if (evpn_prefix.type() != EvpnPrefix::MacAdvertisementRoute &&
311  evpn_prefix.type() != EvpnPrefix::IpPrefixRoute &&
312  evpn_prefix.type() != EvpnPrefix::SelectiveMulticastRoute) {
313  return false;
314  }
315 
316  if (!evpn_prefix.mac_addr().IsBroadcast() &&
317  (evpn_prefix.type() != EvpnPrefix::SelectiveMulticastRoute)) {
318  UpdateInfo *uinfo = GetUpdateInfo(ribout, evpn_route, peerset);
319  if (!uinfo)
320  return false;
321  uinfo_slist->push_front(*uinfo);
322  return true;
323  }
324 
325  if (!evpn_manager_ || evpn_manager_->deleter()->IsDeleted())
326  return false;
327 
328  const IPeer *peer = evpn_route->BestPath()->GetPeer();
329  if (!peer || !ribout->IsRegistered(const_cast<IPeer *>(peer)))
330  return false;
331 
332  size_t peerbit = ribout->GetPeerIndex(const_cast<IPeer *>(peer));
333  if (!peerset.test(peerbit))
334  return false;
335 
336  UpdateInfo *uinfo = evpn_manager_->GetUpdateInfo(evpn_route);
337  if (!uinfo)
338  return false;
339 
340  uinfo->target.set(peerbit);
341  uinfo_slist->push_front(*uinfo);
342  return true;
343 }
