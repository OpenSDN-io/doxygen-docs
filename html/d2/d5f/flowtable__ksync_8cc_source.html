186 void FlowTableKSyncEntry::SetPcapData(FlowEntryPtr fe,
187  std::vector<int8_t> &data) {
188  data.clear();
189  uint32_t addr = ksync_obj_->ksync()->agent()->router_id().to_ulong();
190  data.push_back(FlowEntry::PCAP_CAPTURE_HOST);
191  data.push_back(0x4);
192  data.push_back(((addr >> 24) & 0xFF));
193  data.push_back(((addr >> 16) & 0xFF));
194  data.push_back(((addr >> 8) & 0xFF));
195  data.push_back(((addr) & 0xFF));
196 
197  data.push_back(FlowEntry::PCAP_FLAGS);
198  data.push_back(0x4);
199  uint32_t action;
200  action = fe->match_p().action_info.action;
201  if (fe->is_flags_set(FlowEntry::IngressDir)) {
202  // Set 31st bit for ingress
203  action |= 0x40000000;
204  }
205  data.push_back((action >> 24) & 0xFF);
206  data.push_back((action >> 16) & 0xFF);
207  data.push_back((action >> 8) & 0xFF);
208  data.push_back((action) & 0xFF);
209 
210  data.push_back(FlowEntry::PCAP_SOURCE_VN);
211  data.push_back(fe->data().source_vn_match.size());
212  data.insert(data.end(), fe->data().source_vn_match.begin(),
213  fe->data().source_vn_match.end());
214  data.push_back(FlowEntry::PCAP_DEST_VN);
215  data.push_back(fe->data().dest_vn_match.size());
216  data.insert(data.end(), fe->data().dest_vn_match.begin(),
217  fe->data().dest_vn_match.end());
218  data.push_back(FlowEntry::PCAP_TLV_END);
219  data.push_back(0x0);
220 }

235 int FlowTableKSyncEntry::Encode(sandesh_op::type op, char *buf, int buf_len) {
236  vr_flow_req &req = ksync_obj_->flow_req();
237  int encode_len;
238  int error;
239  uint16_t action = 0;
240  uint16_t drop_reason = VR_FLOW_DR_UNKNOWN;
241 
242  // currently vrouter doesnot guarantee gen id to always start from 0
243  // on vrouter-agent restart
244  // TODO(prabhjot) need to move last gen id seen by vrouter in KSync
245  // Index Manager
246  if (gen_id_ != evict_gen_id_) {
247  // skip sending update to vrouter for evicted entry
248  flow_entry_->LogFlow(FlowEventLog::FLOW_MSG_SKIP_EVICTED, this,
249  hash_id_, evict_gen_id_);
250  return 0;
251  }
252 
253  req.set_fr_op(flow_op::FLOW_SET);
254  req.set_fr_rid(0);
255  req.set_fr_index(hash_id_);
256  req.set_fr_gen_id(gen_id_);
257  const FlowKey *fe_key = &flow_entry_->key();
258  EncodeKSyncIp(&req, fe_key->src_addr, fe_key->dst_addr);
259  req.set_fr_flow_proto(fe_key->protocol);
260  req.set_fr_flow_sport(htons(fe_key->src_port));
261  req.set_fr_flow_dport(htons(fe_key->dst_port));
262  req.set_fr_flow_nh_id(fe_key->nh);
263  if (flow_entry_->key().family == Address::INET)
264  req.set_fr_family(AF_INET);
265  else
266  req.set_fr_family(AF_INET6);
267  req.set_fr_flow_vrf(flow_entry_->data().vrf);
268  uint16_t flags = 0;
269  uint16_t flags1 = 0;
270 
271  if (op == sandesh_op::DEL) {
272  if (hash_id_ == FlowEntry::kInvalidFlowHandle) {
273  return 0;
274  }
275 
276  req.set_fr_flags(0);
277  req.set_fr_flags1(0);
278  // Sync() is not called in case of delete. Copy the event to use
279  // the right token
280  last_event_ = (FlowEvent::Event)flow_entry_->last_event();
281  } else {
282  flags = VR_FLOW_FLAG_ACTIVE;
283  uint32_t fe_action = flow_entry_->match_p().action_info.action;
284  if ((fe_action) & (1 << TrafficAction::PASS)) {
285  action = VR_FLOW_ACTION_FORWARD;
286  }
287 
288  if ((fe_action) & (1 << TrafficAction::DENY)) {
289  action = VR_FLOW_ACTION_DROP;
290  drop_reason = GetDropReason(flow_entry_->data().drop_reason);
291  }
292 
293  if (action == VR_FLOW_ACTION_FORWARD &&
294  flow_entry_->is_flags_set(FlowEntry::NatFlow)) {
295  action = VR_FLOW_ACTION_NAT;
296  }
297 
298  if (action == VR_FLOW_ACTION_NAT &&
299  flow_entry_->reverse_flow_entry() == NULL) {
300  action = VR_FLOW_ACTION_DROP;
301  }
302 
303  if ((fe_action) & (1 << TrafficAction::HBS)) {
304  if (flow_entry_->is_flags_set(FlowEntry::HbfFlow)) {
305  if (flow_entry_->GetHbsInterface() ==
306  FlowEntry::HBS_INTERFACE_RIGHT) {
307  flags1 |= VR_FLOW_FLAG1_HBS_RIGHT;
308  } else {
309  flags1 |= VR_FLOW_FLAG1_HBS_LEFT;
310  }
311  }
312  }
313 
314  if ((fe_action) & (1 << TrafficAction::MIRROR)) {
315  flags |= VR_FLOW_FLAG_MIRROR;
316  req.set_fr_mir_id(-1);
317  req.set_fr_sec_mir_id(-1);
318  if (flow_entry_->match_p().action_info.mirror_l.size() >
319  FlowEntry::kMaxMirrorsPerFlow) {
320  FLOW_TRACE(Err, hash_id_,
321  "Don't support more than two mirrors/analyzers per "
322  "flow:" + integerToString
323  (flow_entry_->
324  data().match_p.action_info.mirror_l.size()));
325  }
326  // Lookup for fist and second mirror entries
327  std::vector<MirrorActionSpec>::const_iterator it;
328  it = flow_entry_->match_p().action_info.mirror_l.begin();
329  MirrorKSyncObject* obj = ksync_obj_->ksync()->agent()->ksync()->
330  mirror_ksync_obj();
331  uint16_t idx_1 = obj->GetIdx((*it).analyzer_name);
332  req.set_fr_mir_id(idx_1);
333  FLOW_TRACE(ModuleInfo, "Mirror index first: " +
334  integerToString(idx_1));
335  ++it;
336  if (it != flow_entry_->match_p().action_info.mirror_l.end()) {
337  uint16_t idx_2 = obj->GetIdx((*it).analyzer_name);
338  if (idx_1 != idx_2) {
339  req.set_fr_sec_mir_id(idx_2);
340  FLOW_TRACE(ModuleInfo, "Mirror index second: " +
341  integerToString(idx_2));
342  } else {
343  FLOW_TRACE(Err, hash_id_,
344  "Both Mirror indexes are same, hence didn't set "
345  "the second mirror dest.");
346  }
347  }
348  req.set_fr_mir_vrf(flow_entry_->data().mirror_vrf);
349  req.set_fr_mir_sip(htonl(ksync_obj_->ksync()->agent()->
350  router_id().to_ulong()));
351  req.set_fr_mir_sport(htons(ksync_obj_->ksync()->agent()->
352  mirror_port()));
353  std::vector<int8_t> pcap_data;
354  SetPcapData(flow_entry_, pcap_data);
355  req.set_fr_pcap_meta_data(pcap_data);
356  }
357 
358  if (flow_entry_->data().component_nh_idx !=
359  (uint32_t)CompositeNH::kInvalidComponentNHIdx) {
360  req.set_fr_ecmp_nh_index(flow_entry_->data().component_nh_idx);
361  } else {
362  req.set_fr_ecmp_nh_index(-1);
363  }
364 
365  if (action == VR_FLOW_ACTION_NAT) {
366  FlowEntry *nat_flow = flow_entry_->reverse_flow_entry();
367  const FlowKey *nat_key = &nat_flow->key();
368 
369  if (flow_entry_->key().src_addr != nat_key->dst_addr) {
370  flags |= VR_FLOW_FLAG_SNAT;
371  }
372  if (flow_entry_->key().dst_addr != nat_key->src_addr) {
373  flags |= VR_FLOW_FLAG_DNAT;
374  }
375 
376  if (flow_entry_->key().protocol == IPPROTO_TCP ||
377  flow_entry_->key().protocol == IPPROTO_UDP) {
378  if (flow_entry_->key().src_port != nat_key->dst_port) {
379  flags |= VR_FLOW_FLAG_SPAT;
380  }
381  if (flow_entry_->key().dst_port != nat_key->src_port) {
382  flags |= VR_FLOW_FLAG_DPAT;
383  }
384  }
385 
386  //Link local, flag determines relaxed policy
387  if (nat_flow->is_flags_set(FlowEntry::LinkLocalBindLocalSrcPort)) {
388  flags |= VR_FLOW_FLAG_LINK_LOCAL;
389  }
390 
391  //Bgp service, flag determines relaxed policy
392  if (nat_flow->is_flags_set(FlowEntry::BgpRouterService)) {
393  flags |= VR_FLOW_BGP_SERVICE;
394  }
395 
396  if (nat_flow->allocated_port()) {
397  flags |= VR_FLOW_BGP_SERVICE;
398  }
399 
400  flags |= VR_FLOW_FLAG_VRFT;
401  req.set_fr_flow_dvrf(flow_entry_->data().dest_vrf);
402  } else if (flow_entry_->is_flags_set(FlowEntry::AliasIpFlow)) {
403  flags |= VR_FLOW_FLAG_VRFT;
404  req.set_fr_flow_dvrf(flow_entry_->data().dest_vrf);
405  }
406 
407  if (fe_action & (1 << TrafficAction::VRF_TRANSLATE)) {
408  flags |= VR_FLOW_FLAG_VRFT;
409  req.set_fr_flow_dvrf(flow_entry_->data().dest_vrf);
410  }
411 
412  if (flow_entry_->is_flags_set(FlowEntry::Trap)) {
413  action = VR_FLOW_ACTION_HOLD;
414  }
415 
416  if (enable_rpf_) {
417  req.set_fr_src_nh_index(src_nh_id_);
418  } else {
419  //Set to discard, vrouter ignores RPF check if
420  //nexthop is set to discard
421  req.set_fr_src_nh_index(0);
422  }
423 
424  FlowEntry *rev_flow = flow_entry_->reverse_flow_entry();
425  if (rev_flow) {
426  flags |= VR_RFLOW_VALID;
427  req.set_fr_rindex(rev_flow->flow_handle());
428  if (rev_flow->flow_handle() == FlowEntry::kInvalidFlowHandle) {
429  const FlowKey &rkey = rev_flow->key();
430  req.set_fr_rflow_nh_id(rkey.nh);
431  uint64_t supper, dupper, slower, dlower;
432 
433  IpToU64(rkey.src_addr, rkey.dst_addr, &supper, &slower,
434  &dupper, &dlower);
435  req.set_fr_rflow_sip_l(slower);
436  req.set_fr_rflow_sip_u(supper);
437  req.set_fr_rflow_dip_l(dlower);
438  req.set_fr_rflow_dip_u(dupper);
439 
440  req.set_fr_rflow_sport(htons(rkey.src_port));
441  req.set_fr_rflow_dport(htons(rkey.dst_port));
442  }
443  }
444 
445  if (flow_entry_->IsShortFlow()) {
446  action = VR_FLOW_ACTION_DROP;
447  }
448 
449  req.set_fr_flags(flags);
450  req.set_fr_flags1(flags1);
451  req.set_fr_action(action);
452  req.set_fr_drop_reason(drop_reason);
453  req.set_fr_qos_id(qos_config_idx);
454  req.set_fr_ttl(flow_entry_->data().ttl);
455  req.set_fr_underlay_ecmp_index(underlay_gw_index);
456  }
457 
458  FlowProto *proto = ksync_obj_->ksync()->agent()->pkt()->get_flow_proto();
459  token_ = proto->GetToken(last_event_);
460  encode_len = req.WriteBinary((uint8_t *)buf, buf_len, &error);
461  return encode_len;
462 }

464 bool FlowTableKSyncEntry::Sync() {
465  bool changed = false;
466 
467  last_event_ = (FlowEvent::Event)flow_entry_->last_event();
468  FlowEntry *rev_flow = flow_entry_->reverse_flow_entry();
469  if (rev_flow) {
470  if (old_reverse_flow_id_ != rev_flow->flow_handle()) {
471  if (old_reverse_flow_id_ != FlowEntry::kInvalidFlowHandle)
472  changed = true;
473  old_reverse_flow_id_ = rev_flow->flow_handle();
474  }
475  }
476 
477  if (flow_entry_->match_p().action_info.action != old_action_) {
478  old_action_ = flow_entry_->match_p().action_info.action;
479  changed = true;
480  }
481 
482  if (flow_entry_->data().drop_reason != old_drop_reason_) {
483  old_drop_reason_ = flow_entry_->data().drop_reason;
484  changed = true;
485  }
486  if (flow_entry_->data().component_nh_idx != old_component_nh_idx_) {
487  old_component_nh_idx_ = flow_entry_->data().component_nh_idx;
488  changed = true;
489  }
490 
491  if (vrouter_gen_id_ != gen_id_) {
492  vrouter_gen_id_ = gen_id_;
493  changed = true;
494  }
495 
496  if (vrouter_hash_id_ != hash_id_) {
497  vrouter_hash_id_ = hash_id_;
498  changed = true;
499  }
500 
501  MirrorKSyncObject* obj = ksync_obj_->ksync()->mirror_ksync_obj();
502  // Lookup for fist and second mirror entries
503  std::vector<MirrorActionSpec>::const_iterator it;
504  it = flow_entry_->match_p().action_info.mirror_l.begin();
505  if (it != flow_entry_->match_p().action_info.mirror_l.end()) {
506  uint16_t idx = obj->GetIdx((*it).analyzer_name);
507  if (!((*it).analyzer_name.empty()) &&
508  (idx == MirrorTable::kInvalidIndex)) {
509  // runn timer to update flow entry
510  ksync_obj_->UpdateUnresolvedFlowEntry(flow_entry_);
511  } else if (old_first_mirror_index_ != idx) {
512  old_first_mirror_index_ = idx;
513  changed = true;
514  }
515  ++it;
516  if (it != flow_entry_->match_p().action_info.mirror_l.end()) {
517  idx = obj->GetIdx((*it).analyzer_name);
518  if (!((*it).analyzer_name.empty()) &&
519  (idx == MirrorTable::kInvalidIndex)) {
520  // run time and to update flow entry;
521  ksync_obj_->UpdateUnresolvedFlowEntry(flow_entry_);
522  } else if (old_second_mirror_index_ != idx) {
523  old_second_mirror_index_ = idx;
524  changed = true;
525  }
526  }
527  }
528 
529  //Trap reverse flow
530  if (trap_flow_ != flow_entry_->is_flags_set(FlowEntry::Trap)) {
531  trap_flow_ = flow_entry_->is_flags_set(FlowEntry::Trap);
532  changed = true;
533  }
534 
535  if (ecmp_ != flow_entry_->is_flags_set(FlowEntry::EcmpFlow)) {
536  ecmp_ = flow_entry_->is_flags_set(FlowEntry::EcmpFlow);
537  changed = true;
538  }
539 
540  if (enable_rpf_ != flow_entry_->data().enable_rpf) {
541  enable_rpf_ = flow_entry_->data().enable_rpf;
542  changed = true;
543  }
544 
545  uint32_t nh_id = NextHopTable::kRpfDiscardIndex;
546  if (flow_entry_->data().rpf_nh.get()) {

667 bool FlowTableKSyncObject::TimerExpiry() {
668  uint16_t count = 0;
669  while (!unresolved_flow_list_.empty() && count < KFlowUnresolvedListYield) {
670  FlowEntryPtr flow = unresolved_flow_list_.front();
671  FlowEntry *flow_entry = flow.get();
672  unresolved_flow_list_.pop_front();
673  flow_entry->SetUnResolvedList(false);
674  count++;
675  if (!flow_entry->deleted()) {
676  FlowProto *proto = ksync()->agent()->pkt()->get_flow_proto();
677  proto->EnqueueUnResolvedFlowEntry(flow.get());
678  }
679  }
680  if (!unresolved_flow_list_.empty())
681  return true;
682  return false;
683 }
