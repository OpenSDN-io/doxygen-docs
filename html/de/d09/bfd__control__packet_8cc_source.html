212 ResultCode ControlPacket::Verify() const {
213  if (multipoint)
214  return kResultCode_InvalidPacket;
215 
216  if (authentication_present)
217  return kResultCode_NotImplemented;
218 
219  if (receiver_discriminator == 0 && state == kUp)
220  return kResultCode_InvalidPacket;
221 
222  if (poll && final)
223  return kResultCode_InvalidPacket;
224 
225  return kResultCode_Ok;
226 }
