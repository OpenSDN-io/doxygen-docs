102 // We want to scan the flow table every 25% of configured ageing time.
103 // Compute number of timer fires needed to scan the flow-table once.
104 uint32_t FlowStatsCollector::TimersPerScan() {
105  uint64_t scan_time_millisec;
106  /* Use Age Time itself as scan-time for flows */
107 
108  // Convert aging-time configured in micro-sec to millisecond
109  scan_time_millisec = flow_age_time_intvl_ / 1000;
110 
111  // Compute time in which we must scan the complete table to honor the
112  // kFlowScanTime
113  scan_time_millisec = (scan_time_millisec * kFlowScanTime) / 100;
114 
115  // Enforce min value on scan-time
116  if (scan_time_millisec < kFlowStatsTimerInterval) {
117  scan_time_millisec = kFlowStatsTimerInterval;
118  }
119 
120  // Number of timer fires needed to scan table once
121  return scan_time_millisec / kFlowStatsTimerInterval;
122 }

124 // Update entries_to_visit_ based on total flows
125 // Timer fires every kFlowScanTime. Its possible that we may not have visited
126 // all entries by the time next timer fires. So, keep accumulating the number
127 // of entries to visit into entries_to_visit_
128 //
129 // A lower-bound and an upper-bound are enforced on entries_to_visit_
130 void FlowStatsCollector::UpdateEntriesToVisit() {
131  // Compute number of flows to visit per scan-time
132  uint32_t count = flow_export_info_list_.size();
133  uint32_t entries = count / timers_per_scan_;
134 
135  // Update number of entries to visit in flow.
136  // The scan for previous timer may still be in progress. So, accmulate
137  // number of entries to visit
138  entries_to_visit_ += entries;
139 
140  // Cap number of entries to visit to 25% of table
141  if (entries_to_visit_ > ((count * kFlowScanTime)/100))
142  entries_to_visit_ = (count * kFlowScanTime)/100;
143 
144  // Apply lower-limit
145  if (entries_to_visit_ < kMinFlowsPerTimer)
146  entries_to_visit_ = kMinFlowsPerTimer;
147 
148  return;

281 void FlowStatsCollector::UpdateVmiTagBasedStats(FlowExportInfo *info,
282  uint64_t bytes, uint64_t pkts) {
283  FlowEntry *flow = info->flow();
284 
285  const Interface *itf = flow->intf_entry();
286  if (!itf) {
287  return;
288  }
289  if (itf->type() != Interface::VM_INTERFACE) {
290  return;
291  }
292  const VmInterface *vmi = static_cast<const VmInterface *>(itf);
293  const string &src_vn = flow->data().source_vn_match;
294  const string &dst_vn = flow->data().dest_vn_match;
295 
296  /* Ignore flows for which source VN or destination VN are not known */
297  if (!src_vn.length() || !dst_vn.length()) {
298  return;
299  }
300 
301  InterfaceUveStatsTable *itf_table = static_cast<InterfaceUveStatsTable *>
302  (agent_uve_->interface_uve_table());
303  EndpointStatsInfo ep;
304  ep.vmi = vmi;
305  ep.local_tagset = flow->local_tagset();
306  ep.remote_tagset = flow->remote_tagset();
307  ep.remote_prefix = flow->RemotePrefix();
308  ep.policy = flow->fw_policy_name_uuid();
309  ep.diff_bytes = bytes;
310  ep.diff_pkts = pkts;
311  FlowTable::GetFlowSandeshActionParams(flow->data().match_p.action_info,
312  ep.action);
313  if (flow->is_flags_set(FlowEntry::LocalFlow)) {
314  /* When VM A talks to VM B which is in different compute nodes, the
315  * following flows are created
316  * (1) A-B, Ingress, Forward, pol1
317  * (2) B-A, Egress, Reverse, pol1
318  * (3) A-B, Egress, Forward, pol2
319  * (4) B-A, Inress, Reverse, pol2
320  * When both A and B are in single compute, we have only the following
321  * flows (Flows marked as LocalFlow)
322  * (1) A-B, Ingress, Forward, pol1
323  * (2) B-A, Inress, Reverse, pol2
324  * To simulate session stats similar to case where VMs are in different
325  * computes, for local flows, we do the following.
326  * (a) when "A-B, Ingress, Forward, pol1" flow is seen, we also
327  * update stats for "A-B, Egress, Forward, pol2". This is because
328  * diff stats for "A-B, Ingress, Forward, pol1" and
329  * "A-B, Egress, Forward, pol2" are same. Policy for implicit flow
330  * is picked from reverse flow
331  * (b) when "B-A, Ingress, Reverse, pol2" flow is seen, we also
332  * update stats for "B-A, Egress, Reverse, pol1". This is because diff
333  * stats for "B-A, Ingress, Reverse, pol2" and
334  * "B-A, Egress, Reverse, pol1" is same. Policy for implicit flow is
335  * picked from reverse flow
336  */
337  ep.local_vn = src_vn;
338  ep.remote_vn = dst_vn;
339  ep.in_stats = true;
340  bool egress_flow_is_client;
341  if (flow->is_flags_set(FlowEntry::ReverseFlow)) {
342  ep.client = false;
343  egress_flow_is_client = true;
344  } else {
345  ep.client = true;
346  egress_flow_is_client = false;
347  }
348  itf_table->UpdateVmiTagBasedStats(ep);
349 
350  /* Local flows will not have egress flows in the system. So we need to
351  * explicitly build stats for egress flow using the data available from
352  * ingress flow. Egress flow stats has to be updated on destination
353  * VMI. We skip updation if we are unable to pick destination VMI from
354  * reverse flow. */
355 
356  FlowEntry* rflow = info->reverse_flow();
357  if (rflow) {
358  const Interface *ritf = rflow->intf_entry();
359  if (ritf && (ritf->type() == Interface::VM_INTERFACE)) {
360  ep.local_tagset = flow->remote_tagset();
361  ep.remote_tagset = flow->local_tagset();
362  ep.local_vn = dst_vn;
363  ep.remote_vn = src_vn;
364  ep.policy = rflow->fw_policy_name_uuid();
365  ep.client = egress_flow_is_client;
366  ep.vmi = static_cast<const VmInterface *>(ritf);
367  ep.in_stats = false;
368  itf_table->UpdateVmiTagBasedStats(ep);
369  }
370  }
371  } else {
372  if (flow->is_flags_set(FlowEntry::IngressDir)) {
373  ep.local_vn = src_vn;
374  ep.remote_vn = dst_vn;
375  ep.in_stats = true;
376  if (flow->is_flags_set(FlowEntry::ReverseFlow)) {
377  ep.client = false;
378  } else {
379  ep.client = true;
380  }
381  } else {
382  ep.local_vn = dst_vn;
383  ep.remote_vn = src_vn;
384  ep.in_stats = false;
385  if (flow->is_flags_set(FlowEntry::ReverseFlow)) {
386  ep.client = true;
387  } else {
388  ep.client = false;
389  }
390  }
391  itf_table->UpdateVmiTagBasedStats(ep);
392  }
393 }

583  // reverse flows
584  if (rev_info) {

587  k_flow_rev = ksync_obj->GetKFlowStats(rfe->key(),
588  rev_info->flow_handle(),
589  rev_info->gen_id(),
590  &k_rflow_stats);
591  if (ShouldBeAged(rev_info, k_flow_rev, k_rflow_stats, curr_time)) {
592  deleted = true;
593  }
594  } else {
595  deleted = true;
596  }
597  }
598 
599  if (deleted == true) {
600  FlowDeleteEnqueue(info, curr_time);
601  }
602 
603  // Update stats for flows not being deleted
604  // Stats for deleted flow are updated when we get DELETE message
605  if (deleted == false && k_flow) {

607 
608  k_bytes = GetFlowStats(k_stats.flow_bytes_oflow,
609  k_stats.flow_bytes);
610  bytes = 0x0000ffffffffffffULL & info->bytes();
611  /* Don't account for agent overflow bits while comparing change in
612  * stats */
613  if (bytes != k_bytes) {
614  UpdateFlowStatsInternalLocked(info,
615  k_stats.flow_bytes,
616  k_stats.flow_bytes_oflow,
617  k_stats.flow_packets,
618  k_stats.flow_packets_oflow,
619  curr_time, false);
620  }
621  }

625 // Check if a flow is to be aged or evicted. Returns number of flows visited
626 uint32_t FlowStatsCollector::ProcessFlow(FlowExportInfoList::iterator &it,
627  KSyncFlowMemory *ksync_obj,
628  FlowExportInfo *info,
629  uint64_t curr_time) {
630  uint32_t count = 1;
631  FlowEntry *fe = info->flow();
632  /* Use flow-handle and gen-id from FlowExportInfo instead of FlowEntry.
633  * The stats that FlowExportInfo holds corresponds to a given
634  * (FlowKey, gen-id and FlowHandle). Since gen-id/flow-handle for a flow
635  * can change dynamically, we need to pick gen-id and flow-handle from
636  * FlowExportInfo. Otherwise stats will go wrong. Whenever gen-id/
637  * flow-handle changes, the stats will be reset as part of AddFlow API
638  */
639  uint32_t flow_handle = info->flow_handle();
640  uint16_t gen_id = info->gen_id();
641 
642  /* If Flow handle is still not populated in FlowStatsCollector, pick the
643  * value from FlowEntry
644  */
645  if (flow_handle == FlowEntry::kInvalidFlowHandle) {
646  {
647  FlowEntry *rflow = NULL;
648  FLOW_LOCK(fe, rflow, FlowEvent::FLOW_MESSAGE);
649  // since flow processing and stats collector can run in parallel
650  // flow handle and gen id not being the key for flow entry can
651  // change while processing, so flow handle and gen id should be
652  // fetched by holding an lock.
653  flow_handle = fe->flow_handle();
654  gen_id = fe->gen_id();
655  info->CopyFlowInfo(fe);
656  }
657  }
658  const vr_flow_entry *k_flow = NULL;
659  vr_flow_stats k_stats;
660  KFlowData kinfo;
661 
662  /* Teardown time is set when Evicted flow stats update message is received.
663  * For flows whose teardown time is set, we need not read stats from
664  * vrouter
665  */
666  if (!info->teardown_time()) {
667  k_flow = ksync_obj->GetKFlowStatsAndInfo(fe->key(), flow_handle,
668  gen_id, &k_stats, &kinfo);
669 
670  // Flow evicted?
671  if (EvictFlow(ksync_obj, k_flow, kinfo.flags, flow_handle, gen_id,
672  info, curr_time) == true) {
673  // If retry_delete_ enabled, dont change flow_export_info_list_
674  if (retry_delete_ == true)
675  return count;
676 
677  // We dont want to retry delete-events, remove flow from ageing list
678  assert(info->is_linked());
679  FlowExportInfoList::iterator flow_it =
680  flow_export_info_list_.iterator_to(*info);
681  flow_export_info_list_.erase(flow_it);
682 
683  return count;
684  }
685  }
686 
687 
688  // Flow aged?
689  if (AgeFlow(ksync_obj, k_flow, k_stats, kinfo, info, curr_time) == false)
690  return count;
691 
692  // If retry_delete_ enabled, dont change flow_export_info_list_
693  if (retry_delete_ == false)
694  return count;
695 
696  // Flow aged, remove both forward and reverse flow
697  assert(info->is_linked());
698  FlowExportInfoList::iterator flow_it =
699  flow_export_info_list_.iterator_to(*info);
700  flow_export_info_list_.erase(flow_it);
701 
702  FlowEntry *rfe = info->reverse_flow();
703  FlowExportInfo *rev_info = FindFlowExportInfo(rfe);
704  if (rev_info) {
705  if (rev_info->is_linked()) {
706  FlowExportInfoList::iterator rev_flow_it =
707  flow_export_info_list_.iterator_to(*rev_info);
708  if (rev_flow_it == it) {
709  it++;
710  }
711  flow_export_info_list_.erase(rev_flow_it);
712  }
713  count++;
714  }
715  return count;
716 }

718 uint32_t FlowStatsCollector::RunAgeing(uint32_t max_count) {
719  FlowExportInfoList::iterator it;
720  if (flow_iteration_key_ == NULL) {
721  it = flow_export_info_list_.begin();
722  } else {
723  FlowEntryTree::iterator tree_it = flow_tree_.find(flow_iteration_key_);
724  // Flow to iterate next is not found. Force stop this iteration.
725  // We will continue from begining on next timer
726  if (tree_it == flow_tree_.end()) {
727  flow_iteration_key_ = NULL;
728  return entries_to_visit_;
729  }
730  it = flow_export_info_list_.iterator_to(tree_it->second);
731  }
732 
733  KSyncFlowMemory *ksync_obj = agent_uve_->agent()->ksync()->
734  ksync_flow_memory();
735  uint64_t curr_time = GetCurrentTime();
736  uint32_t count = 0;
737  while (count < max_count) {
738  if (it == flow_export_info_list_.end()) {
739  break;
740  }
741 
742  FlowExportInfo *info = &(*it);
743  it++;
744  flows_visited_++;
745  count += ProcessFlow(it, ksync_obj, info, curr_time);
746  }
747 
748  // Update iterator for next pass
749  if (it == flow_export_info_list_.end()) {
750  flow_iteration_key_ = NULL;
751  } else {
752  flow_iteration_key_ = it->flow();
753  }
754 
755  return count;
756 }

850 bool FlowStatsCollector::RequestHandler(boost::shared_ptr<FlowExportReq> req) {
851  const FlowExportInfo &info = req->info();
852  FlowEntry *flow = info.flow();
853  FlowEntry *rflow = info.reverse_flow();
854  FLOW_LOCK(flow, rflow, FlowEvent::FLOW_MESSAGE);
855 
856  switch (req->event()) {
857  case FlowExportReq::ADD_FLOW: {
858  AddFlow(req->info());
859  break;
860  }
861 
862  case FlowExportReq::DELETE_FLOW: {
863  FlowEntryTree::iterator it;
864  // Get the FlowExportInfo for flow
865  if (FindFlowExportInfo(flow, it) == false)
866  break;
867 
868  /* We don't export flows in TSN mode */
869  if (agent_uve_->agent()->tsn_enabled() == false) {
870  FlowExportInfo *info = &it->second;
871  /* While updating stats for evicted flows, we set the teardown_time
872  * and export the flow. So delete handling for evicted flows need
873  * not update stats and export flow */
874  if (!info->teardown_time()) {
875  UpdateFlowStats(info, req->time());
876  }
877  }
878  /* Remove the entry from our tree */
879  DeleteFlow(it);
880  break;
881  }
882 
883  case FlowExportReq::UPDATE_FLOW_STATS: {
884  /* We don't export flows in TSN mode */
885  if (agent_uve_->agent()->tsn_enabled() == false) {
886  EvictedFlowStatsUpdate(flow, req->bytes(), req->packets(),
887  req->oflow_bytes(), req->uuid());
888  }
889  break;
890  }
891 
892  default:
893  assert(0);
894  }
895 
896  if (deleted_ && parent_->CanDelete()) {
897  flow_stats_manager_->Free(flow_aging_key_);
898  }
899 
900  return true;
901 }

968 void FlowStatsCollector::AddFlow(FlowExportInfo info) {
969  /* Before inserting update the gen_id and flow_handle in FlowExportInfo.
970  * Locks for accessing fields of flow are taken in calling function.
971  */
972  FlowEntry* fe = info.flow();
973  info.CopyFlowInfo(fe);
974  std::pair<FlowEntryTree::iterator, bool> ret =
975  flow_tree_.insert(make_pair(fe, info));
976  if (ret.second == false) {
977  FlowExportInfo &prev = ret.first->second;
978  if (prev.uuid() != fe->uuid()) {
979  /* Received ADD request for already added entry with a different
980  * UUID. Because of state-compression of messages to
981  * FlowStatsCollector in FlowMgmt, we have not received DELETE for
982  * previous UUID. Send FlowExport to indicate delete for the flow.
983  * This export need not be sent if teardown time is already set.
984  * Teardown time would be set if EvictedFlowStats update request
985  * comes before this duplicate add.
986  */
987  if (!prev.teardown_time()) {
988  UpdateFlowStats(&prev, info.last_modified_time());
989  }
990  /* After sending Delete to collector (if required), reset the stats
991  */
992  prev.ResetStats();
993  }
994  prev.CopyFlowInfo(fe);
995  prev.set_delete_enqueue_time(0);
996  prev.set_evict_enqueue_time(0);
997  prev.set_teardown_time(0);
998  } else {
999  NewFlow(info.flow());
1000  }
1001  if (ret.first->second.is_linked() == false) {
1002  flow_export_info_list_.push_back(ret.first->second);
1003  }
1004 }
