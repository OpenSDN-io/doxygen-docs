57 //
58 // Insert given path and redo path selection.
59 //
60 void BgpRoute::InsertPath(BgpPath *path) {
61  assert(!IsDeleted());
62  const Path *prev_front = front();
63 
64  BgpTable *table = static_cast<BgpTable *>(get_table());
65  // 'table' is not expected to be null, suspect it is checked here because of
66  // unit tests.
67  if (table) {
68  // Default Tunnel Encapsulation processing is done if configured for
69  // the address family on the peer.
70  // If routing policy is supported on the table(family)it is handled in
71  // ProcessRoutingPolicy where other changes may be made to the
72  // attributes. Also, when there are routing policy configuration changes
73  // the attributes needs to be re-evaluated using the original attributes
74  // and Default Tunnel Encapsulation also needs to be re-applied, this
75  // will be handled seamlessly since ProcessRoutingPolicy will be called
76  // in that case.
77  // Note that currently routing policies are not supported on VPN address
78  // families, however this is ensuring that it will be handed correctly
79  // in future. Additionally, for labeled inet routes, routing policies
80  // are supported so the Default Tunnel Encapsulation needs to be handled
81  // in ProcessRoutingPolicy.
82  // If the table(family) does not support routing policies the Default
83  // Tunnel Encapsulation processing is handled here. The original and
84  // modified attributes are saved and are not expected to be modified
85  // further. Note that the configuration is not applied if the path
86  // already has tunnel encapsulation specified.
87 
88  if (table->IsRoutingPolicySupported()) {
89  if (!path->IsReplicated()) {
90  // Add sub-cluster extended community to all routes
91  // originated within a sub-cluster
92  uint32_t subcluster_id = SubClusterId();
93  if (subcluster_id) {
94  path->AddExtCommunitySubCluster(subcluster_id);
95  }
96  }
97  RoutingInstance *rtinstance = table->routing_instance();
98  rtinstance->ProcessRoutingPolicy(this, path);
99  } else {
100  IPeer *peer = path->GetPeer();
101  if (peer) {
102  // Take snapshot of original attribute
103  BgpAttr *out_attr = new BgpAttr(*(path->GetOriginalAttr()));
104  peer->ProcessPathTunnelEncapsulation(path, out_attr,
105  table->server()->extcomm_db(), table);
106  BgpAttrPtr modified_attr =
107  table->server()->attr_db()->Locate(out_attr);
108  // Update the path with new set of attributes
109  path->SetAttr(modified_attr, path->GetOriginalAttr());
110  }
111  }
112  }
113  insert(path);
114 
115  Sort(&BgpTable::PathSelection, prev_front);
116 
117  // Update counters.
118  if (table)
119  table->UpdatePathCount(path, +1);
120  path->UpdatePeerRefCount(+1, table ? table->family() : Address::UNSPEC);
121 }

123 //
124 // Delete given path and redo path selection.
125 //
126 void BgpRoute::DeletePath(BgpPath *path) {
127  const Path *prev_front = front();
128 
129  remove(path);
130  Sort(&BgpTable::PathSelection, prev_front);
131 
132  // Update counters.
133  BgpTable *table = static_cast<BgpTable *>(get_table());
134  if (table)
135  table->UpdatePathCount(path, -1);
136  path->UpdatePeerRefCount(-1, table ? table->family() : Address::UNSPEC);
137 
138  delete path;
139 }

143 // Skips secondary, aliased and resolved paths.

145 const BgpPath *BgpRoute::FindPath(BgpPath::PathSource src) const {
146  for (Route::PathList::const_iterator it = GetPathList().begin();
147  it != GetPathList().end(); ++it) {
148  // Skip secondary paths.
149  if (dynamic_cast<const BgpSecondaryPath *>(it.operator->())) {
150  continue;
151  }
152 
153  const BgpPath *path = static_cast<const BgpPath *>(it.operator->());
154  if (path->GetFlags() & BgpPath::AliasedPath) {
155  continue;
156  }
157  if (path->GetFlags() & BgpPath::ResolvedPath) {
158  continue;
159  }
160  if (path->GetSource() == src) {
161  return path;
162  }
163  }
164  return NULL;

342 //
343 // Check if there's a better path with the same forwarding information.
344 //
345 // Return true if we find such a path, false otherwise.
346 //
347 // The forwarding information we look at is just the next hop. We don't
348 // consider the label since there's could be transient cases where we
349 // have 2 paths with the same next hop and different labels. We don't
350 // want to treat these as unique paths.
351 //
352 bool BgpRoute::DuplicateForwardingPath(const BgpPath *in_path) const {
353  for (Route::PathList::const_iterator it = GetPathList().begin();
354  it != GetPathList().end(); ++it) {
355  const BgpPath *path = static_cast<const BgpPath *>(it.operator->());
356 
357  // Bail if we reached the input path since the paths are sorted.
358  if (path == in_path)
359  return false;
360 
361  // Check the forwarding information.
362  if (path->GetAttr()->nexthop() == in_path->GetAttr()->nexthop())
363  return true;
364  }
365 
366  return false;
367 }

369 //
370 // Find the secondary path matching secondary replicated info.
371 //
372 BgpPath *BgpRoute::FindSecondaryPath(BgpRoute *src_rt,
373  BgpPath::PathSource src, const IPeer *peer, uint32_t path_id) {
374  for (Route::PathList::iterator it = GetPathList().begin();
375  it != GetPathList().end(); ++it) {
376  BgpSecondaryPath *path = dynamic_cast<BgpSecondaryPath *>(
377  it.operator->());
378  // Skip resolved paths. They'll be taken care of by the path
379  // resolver infrastructure.
380  if (path && (src_rt->table()->IsVpnTable()) &&
381  (path->GetFlags() & BgpPath::ResolvedPath)) {
382  continue;
383  }
384  if (path && path->src_rt() == src_rt &&
385  path->GetPeer() == peer && path->GetPathId() == path_id &&
386  path->GetSource() == src) {
387  return path;
388  }
389  }
390  return NULL;
391 }

650 void BgpRoute::FillRouteInfo(const BgpTable *table,
651  ShowRoute *show_route, const string &source, const string &protocol) const {
652  const RoutingInstance *ri = table->routing_instance();
653 
654  show_route->set_prefix(ToString());
655  show_route->set_last_modified(
656  integerToString(UTCUsecToPTime(last_change_at())));
657 
658  vector<ShowRoutePath> show_route_paths;
659  for (Route::PathList::const_iterator it = GetPathList().begin();
660  it != GetPathList().end(); ++it) {
661  const BgpPath *path = static_cast<const BgpPath *>(it.operator->());
662  ShowRoutePath srp;
663  const IPeer *peer = path->GetPeer();
664 
665  // Filter against peer source, if specified.
666  if (!source.empty() && (!peer || source != peer->ToString()))
667  continue;
668 
669  // Filter against path protocol, if specified.
670  if (!protocol.empty() && protocol != path->GetSourceString())
671  continue;
672 
673  if (peer) {
674  srp.set_source(peer->ToString());
675  }
676 
677  const BgpPeer *bgp_peer = dynamic_cast<const BgpPeer *>(peer);
678  if (bgp_peer) {
679  srp.set_local_as(bgp_peer->local_as());
680  srp.set_peer_as(bgp_peer->peer_as());
681  srp.set_peer_router_id(bgp_peer->bgp_identifier_string());
682  }
683 
684  const BgpAttr *attr = path->GetAttr();
685  if (attr->edge_forwarding()) {
686  FillEdgeForwardingInfo(attr->edge_forwarding(), &srp);
687  }
688  if (attr->edge_discovery()) {
689  FillEdgeDiscoveryInfo(attr->edge_discovery(), &srp);
690  }
691  if (attr->sub_protocol().empty()) {
692  srp.set_protocol(path->GetSourceString());
693  } else {
694  const string sbp = attr->sub_protocol();
695  srp.set_protocol(path->GetSourceString() + " (" + sbp + ")");
696  }
697  srp.set_origin(attr->origin_string());
698  if (attr->as_path() != NULL)
699  srp.set_as_path(attr->as_path()->path().ToString());
700  else if (attr->aspath_4byte() != NULL)
701  srp.set_as_path(attr->aspath_4byte()->path().ToString());
702  if (attr->as4_path() != NULL)
703  srp.set_as4_path(attr->as4_path()->path().ToString());
704  srp.set_local_preference(attr->local_pref());
705  srp.set_med(attr->med());
706  srp.set_next_hop(attr->nexthop().to_string());
707  srp.set_label(path->GetLabel());
708  srp.set_flags(path->GetFlagsStringList());
709  srp.set_last_modified(
710  integerToString(UTCUsecToPTime(path->time_stamp_usecs())));
711  if (path->IsReplicated()) {
712  const BgpSecondaryPath *replicated;
713  replicated = static_cast<const BgpSecondaryPath *>(path);
714  srp.set_replicated(true);
715  srp.set_primary_table(replicated->src_table()->name());
716  } else {
717  srp.set_replicated(false);
718  Address::Family vpn_family =
719  Address::VpnFamilyFromFamily(table->family());
720  const RoutePathReplicator *replicator =
721  table->server()->replicator(vpn_family);
722  if (replicator) {
723  srp.set_secondary_tables(
724  replicator->GetReplicatedTableNameList(table, this, path));
725  }
726  }
727  if (attr->cluster_list()) {
728  FillRoutePathClusterListInfo(attr->cluster_list(), &srp);
729  }
730  vector<string> communities;
731  if (attr->community()) {
732  FillRoutePathCommunityInfo(attr->community(), &srp, &communities);
733  }
734  if (attr->ext_community()) {
735  FillRoutePathExtCommunityInfo(table, attr->ext_community(), &srp,
736  &communities);
737  }
738  if (attr->large_community()) {
739  FillRoutePathLargeCommunityInfo(attr->large_community(), &srp,
740  &communities);
741  }
742  if (!communities.empty()){
743  srp.set_communities(communities);
744  }
745  if (srp.get_origin_vn().empty() &&
746  !table->IsVpnTable() && path->IsVrfOriginated()) {
747  srp.set_origin_vn(ri->GetVirtualNetworkName());
748  }
749  if (attr->origin_vn_path()) {
750  FillOriginVnPathInfo(attr->origin_vn_path(), &srp);
751  }
752  if (attr->pmsi_tunnel()) {
753  const ExtCommunity *extcomm = attr->ext_community();
754  if (extcomm) {
755  FillPmsiTunnelInfo(attr->pmsi_tunnel(), extcomm, &srp);
756  }
757  }
758  if (attr->originator_id().to_ulong()) {
759  srp.set_originator_id(attr->originator_id().to_string());
760  }
761 
762  show_route_paths.push_back(srp);
763  }
764  show_route->set_paths(show_route_paths);
