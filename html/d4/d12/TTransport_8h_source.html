35 int32_t readAll(Transport_ &trans, uint8_t* buf, uint32_t len) {
36  uint32_t have = 0;
37  uint32_t get = 0;
38 
39  while (have < len) {
40  get = trans.read(buf+have, len-have);
41  if (get <= 0) {
42  LOG(ERROR, __func__ << ": No more data to read");
43  return -1;
44  }
45  have += get;
46  }
47 
48  return have;
49 }
