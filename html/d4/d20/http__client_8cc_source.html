33 void HttpClientSession::OnEventInternal(TcpSessionPtr session, Event event) {
34  if (event_cb_ && !event_cb_.empty()) {
35  event_cb_(static_cast<HttpClientSession *>(session.get()), event);
36  }
37 
38  if (event == CLOSE) {
39  goto error;
40  }
41  if (event == ACCEPT) {
42  goto error;
43  }
44  if (event == CONNECT_COMPLETE) {
45  goto error;
46  }
47  if (event == CONNECT_FAILED) {
48  goto error;
49  }
50  if (event == EVENT_NONE) {
51  goto error;
52  }
53  return;
54 
55 error:
56  // Call callback function with error;
57  return;
58 }

214 void HttpConnection::HttpProcessInternal(const std::string body,
215  std::string path,
216  unsigned short hdr_shortTimeout_reuse,
217  std::vector<std::string> hdr_options,
218  HttpCb cb, http_method method) {
219  bool short_timeout, reuse;
220  bf2bool(hdr_shortTimeout_reuse, sent_hdr_, short_timeout, reuse);
221  state_ = STATUS;
222  status_ = 0;
223  version_.clear();
224  reason_.clear();
225  if (client()->AddConnection(this) == false) {
226  // connection already exists
227  if (!reuse)
228  return;
229  }
230 
231  struct _GlobalInfo *gi = client()->GlobalInfo();
232  struct _ConnInfo *curl_handle = new_conn(this, gi, sent_hdr_, short_timeout,
233  reuse);
234  if (!curl_handle) {
235  LOG(DEBUG, "Http : unable to create new connection");
236  return;
237  }
238 
239  if (curl_handle_) {
240  // delete existing curl_handle
241  del_curl_handle(curl_handle_, gi);
242  }
243  curl_handle->connection = this;
244  set_curl_handle(curl_handle);
245 
246  cb_ = cb;
247 
248  std::string url = make_url(path);
249  set_url(curl_handle_, url.c_str());
250 
251  // set special curl options requested in http connection
252  std::map<CURLoption, int> *curl_options = this->curl_options();
253  std::map<CURLoption, int>::iterator iter = curl_options->begin();
254  while (iter != curl_options->end()) {
255  set_curl_option(curl_handle_->easy, iter->first, iter->second);
256  iter++;
257  }
258 
259  // set SSL curl options
260  if (use_ssl_) {
261  set_ssl_options(curl_handle_, client_cert_.c_str(),
262  client_cert_type_.c_str(), client_key_.c_str(), ca_cert_.c_str());
263  }
264 
265  // Add header options to the get request
266  for (uint32_t i = 0; i < hdr_options.size(); ++i)
267  set_header_options(curl_handle_, hdr_options[i].c_str());
268 
269  switch (method) {
270  case HTTP_GET:
271  http_get(curl_handle_, gi);
272  break;
273 
274  case HTTP_HEAD:
275  http_head(curl_handle_, gi);
276  break;
277 
278  case HTTP_POST:
279  if (!hdr_options.size()) {
280  // if no header options are set, set the content type
281  set_header_options(curl_handle_, "Content-Type: application/xml");
282  }
283  set_post_string(curl_handle_, body.c_str(), body.size());
284  http_post(curl_handle_, gi);
285  break;
286 
287  case HTTP_PUT:
288  if (!hdr_options.size()) {
289  // if no header options are set, set the content type
290  set_header_options(curl_handle_, "Content-Type: application/xml");
291  }
292  set_put_string(curl_handle_, body.c_str(), body.size());
293  http_put(curl_handle_, gi);
294  break;
295 
296  case HTTP_DELETE:
297  http_delete(curl_handle_, gi);
298  break;
299 
300  default:
301  assert(0);
302  }
303 }

315 void HttpConnection::AssignHeader(const char *ptr, size_t size) {
316 
317  buf_.assign(ptr, size);
318 
319  switch (state_) {
320  case STATUS: {
321  status_ = 0;
322  int i = buf_.find(' ', 0);
323  version_ = boost::algorithm::trim_copy(buf_.substr(0, i));
324  int j = buf_.find(' ', i+1);
325  status_ = atoi(buf_.substr(i+1, j).c_str());
326  reason_ = boost::algorithm::trim_copy(buf_.substr(j+1));
327 #ifdef __DEBUG__
328  //for (std::string::iterator ii=buf_.begin()+i+1;
329  // ii != buf_.begin()+j; ii++) {
330  // status_ = (status_ << 3) + (status_ << 1) + (*ii - '0');
331  //}
332  std::cout << "Status Line: " << std::dec << status_ << ":"
333  << reason_ << "(" << version_ << ")" << std::endl;
334 #endif
335  state_ = HEADER;
336  break;
337  }
338  case HEADER:
339  if (buf_ != "\r\n") {
340  std::istringstream iss(buf_);
341  std::string tok;
342  while (std::getline(iss, tok, '\r') && tok != "\n") {
343  unsigned int i = tok.find(':', 0);
344  if (i != std::string::npos) {
345  headers_.insert(std::make_pair(
346  boost::algorithm::trim_copy(tok.substr(0, i)),
347  boost::algorithm::trim_copy(tok.substr(i + 1))));
348  }
349  }
350  }
351  break;
352  }
353  boost::system::error_code error;
354  // callback to client *backward compatibility*
355  if (sent_hdr_ && cb_ != NULL) {
356  cb_(buf_, error);
357  }
358 }
