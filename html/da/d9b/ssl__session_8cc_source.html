33 class SslSession::SslReader : public Task {
34 public:
35  typedef function<void(Buffer)> ReadHandler;
36 
37  SslReader(int task_id, SslSessionPtr session, ReadHandler read_fn,
38  Buffer buffer)
39  : Task(task_id, session->GetSessionInstance()),
40  session_(session), read_fn_(read_fn), buffer_(buffer) {
41  }
42  virtual bool Run() {
43  if (session_->IsEstablished()) {
44  session_->ssl_last_read_len_ = BufferSize(buffer_);
45  read_fn_(buffer_);
46  if (session_->IsReaderDeferred()) {
47  // Update socket read block count.
48  session_->stats_.read_block_start_time = UTCTimestampUsec();
49  session_->stats_.read_blocked++;
50  session_->server()->stats_.read_blocked++;
51  } else {
52  if (session_->IsSslDisabled()) {
53  session_->AsyncReadStart();
54  } else if (!session_->IsSslHandShakeInProgress()) {
55  session_->AsyncReadStart();
56  }
57  }
58  }
59  return true;
60  }
61  string Description() const { return "SslSession::SslReader"; }
62 
63 private:
64  SslSessionPtr session_;
65  ReadHandler read_fn_;
66  Buffer buffer_;
67 };

127 //
128 // Check if a socker error is hard and fatal. Only then should we close the
129 // socket. Soft errors like EINTR and EAGAIN should be ignored or properly
130 // handled with retries
131 //
132 bool SslSession::IsSocketErrorHard(const error_code &ec) {
133 
134  bool error;
135  error = TcpSession::IsSocketErrorHard(ec);
136 #if defined(SSL_R_SHORT_READ) // openssl 1.0
137  if (ec.value() == ERR_PACK(ERR_LIB_SSL, 0, SSL_R_SHORT_READ)) {
138  error = false;
139  }
140 #elif defined(BOOST_ASIO_SSL_ERROR_STREAM_TRUNCATED) // new openssl and boost
141  if (ec.value() == boost::asio::ssl::error::stream_truncated) {
142  error = false;
143  }
144 #endif
145 
146  return error;
147 }

168 void SslSession::SslHandShakeCallback(SslHandShakeCallbackHandler cb,
169  SslSessionPtr session,
170  const error_code &error) {
171  session->ssl_handshake_in_progress_ = false;
172  if (!error) {
173  session->SetSslHandShakeSuccess();
174  } else {
175  session->SetSslHandShakeFailure();
176  }
177 
178  if (session->socket() != NULL && !(session->IsClosed())) {
179  cb(session, error);
180  }

183 void SslSession::TriggerSslHandShakeInternal(
184  SslSessionPtr session, SslHandShakeCallbackHandler cb) {
185  srand(static_cast<unsigned>(time(0)));
186  error_code ec;
187  session->ssl_handshake_in_progress_ = true;
188  if (session->IsServerSession()) {
189  session->ssl_socket_->async_handshake(stream_base::server,
190  bind(&SslSession::SslHandShakeCallback, cb, session,
191  error));
192  } else {
193  session->ssl_socket_->async_handshake(stream_base::client,
194  bind(&SslSession::SslHandShakeCallback, cb, session,
195  error));
196  }
