54  static std::string escapeXMLControlCharsInternal(const std::string& str) {
55  std::string xmlstr;
56  xmlstr.reserve(str.length());
57  for (std::string::const_iterator it = str.begin();
58  it != str.end(); ++it) {
59  switch(*it) {
60  case '&': xmlstr += "&amp;"; break;
61  case '\'': xmlstr += "&apos;"; break;
62  case '<': xmlstr += "&lt;"; break;
63  case '>': xmlstr += "&gt;"; break;
64  default: xmlstr += *it;
65  }
66  }
67  return xmlstr;
68  }
