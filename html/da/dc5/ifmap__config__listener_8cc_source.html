36 // We register one listener for the IFMapLinkTable and a listener for each of

39 // Pure virtual DependencyTrackerInit() should initialize dependency tracker's
40 // policy_map, which is then used to register listeners for specified tables.
41 //
42 void IFMapConfigListener::Initialize() {
43  DB *database = manager_->database();
44 
45  tracker_.reset(
46  new IFMapDependencyTracker(
47  database, manager_->graph(),
48  boost::bind(&IFMapConfigListener::ChangeListAdd, this, _1)));

53  assert(link_table != NULL);

56  boost::bind(&IFMapConfigListener::LinkObserver, this, _1, _2));

109 //
110 // Add an IFMapNode to the ChangeList by creating a ConfigDelta.
111 //
112 // We take references on the IFMapNode and the IFMapObject.
113 //
114 void IFMapConfigListener::ChangeListAdd(IFMapNode *node) {
115  CHECK_CONCURRENCY(kConcurrency_.c_str(), "db::DBTable", "db::IFMapTable");
116 
117  IFMapTable *table = node->table();
118  TableMap::const_iterator tid = table_map_.find(table->name());
119  if (tid == table_map_.end()) {
120  return;
121  }
122 
123  ConfigDelta delta;
124  delta.id_type = table->Typename();
125  delta.id_name = node->name();
126  if (!node->IsDeleted()) {
127  DBState *current = node->GetState(table, tid->second);
128  if (current == NULL) {
129  delta.node = IFMapNodeRef(new IFMapNodeProxy(node, tid->second));
130  }
131  delta.obj = IFMapObjectRef(node->GetObject());
132  }
133  change_list_.push_back(delta);
134 }

141 void IFMapConfigListener::NodeObserver(
142  DBTablePartBase *root, DBEntryBase *db_entry) {
143  CHECK_CONCURRENCY("db::DBTable", "db::IFMapTable");
144 
145  // Ignore deleted nodes for which the configuration code doesn't hold
146  // state. This is the case with ie. BgpRouter objects other than the local
147  // node.
148  IFMapNode *node = static_cast<IFMapNode *>(db_entry);
149  if (node->IsDeleted()) {
150  IFMapTable *table = node->table();
151  TableMap::const_iterator tid = table_map_.find(table->name());
152  assert(tid != table_map_.end());
153  if (node->GetState(table, tid->second) == NULL) {
154  return;
155  }
156  }
157 
158  tracker_->NodeEvent(node);
159  manager_->OnChange();
160 }
