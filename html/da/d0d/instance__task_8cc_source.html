62 // If there is an error before the fork, task is set to "not running"
63 // and "false" is returned to caller so that caller can take appropriate
64 // action on task. If an error is encounted after fork, it is very
65 // likely that child process is running so we keep the task status as
66 // "running" and return "false" to caller, so that caller does not
67 // attempt to run the same task again. In this case, the child process
68 // exit notification can not be received by instance manager, hence
69 // instance manager has to rely on TaskTimeout delete the task.
70 bool InstanceTaskExecvp::Run() {
71  std::vector<std::string> argv;
72 
73  is_running_ = true;
74 
75  boost::split(argv, cmd_, boost::is_any_of(" "), boost::token_compress_on);
76  std::vector<const char *> c_argv(argv.size() + 1);
77  for (std::size_t i = 0; i != argv.size(); ++i) {
78  c_argv[i] = argv[i].c_str();
79  }
80 
81  int err[2];
82  if (pipe(err) < 0) {
83  return is_running_ = false;
84  }
85 
86  pid_ = vfork();
87  if (pid_ == 0) {
88  close(err[0]);
89  if (pipe_stdout_) {
90  dup2(err[1], STDOUT_FILENO);
91  } else {
92  dup2(err[1], STDERR_FILENO);
93  }
94  close(err[1]);
95 
96  if (!pipe_stdout_) {
97  close(STDOUT_FILENO);
98  close(STDIN_FILENO);
99  }
100 
101  /* Close all the open fds before execvp */
102  CloseTaskFds();
103  execvp(c_argv[0], (char **) c_argv.data());
104  perror("execvp");
105 
106  _exit(127);
107  }
108 
109  close(err[1]);
110 
111  start_time_ = time(NULL);
112 
113  int fd = ::dup(err[0]);
114  close(err[0]);
115  if (fd == -1) {
116  //is_running_ is still true indicating the child process is
117  //running. Caller needs to verify the status before acting on
118  //the task again
119  return false;
120  }
121  boost::system::error_code ec;
122  input_.assign(fd, ec);
123  if (ec) {
124  close(fd);
125 
126  //is_running_ is still true indicating the child process is
127  //running. Caller needs to verify the status before acting on
128  //the task again
129  return false;
130  }
131  setup_done_ = true;
132 
133  bzero(rx_buff_, sizeof(rx_buff_));
134  input_.async_read_some(boost::asio::buffer(rx_buff_, kBufLen),
135  boost::bind(&InstanceTaskExecvp::ReadData,
136  this, boost::asio::placeholders::error,
137  boost::asio::placeholders::bytes_transferred));
138  return true;
139 
140 }
