95  void set_session(SandeshSession * session, bool enq) {
96  session = session_.exchange(session);
97  if (session != NULL) {

99  session->SetReceiveMsgCb(NULL);
100  session->Close();
101  session->Shutdown();

103  EnqueDelSession(session);
104  }
105  }
