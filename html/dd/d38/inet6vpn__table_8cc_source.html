68 BgpRoute *Inet6VpnTable::RouteReplicate(BgpServer *server, BgpTable *src_table,
69  BgpRoute *source_rt, const BgpPath *src_path,
70  ExtCommunityPtr community) {
71  assert(src_table->family() == Address::INET6);
72 
73  Inet6Route *src_rt = dynamic_cast<Inet6Route *>(source_rt);
74  assert(src_rt);
75 
76  const RouteDistinguisher &rd = GenerateDistinguisher(src_table, src_path);
77  Inet6VpnPrefix vpn_prefix(rd, src_rt->GetPrefix().ip6_addr(),
78  src_rt->GetPrefix().prefixlen());
79 
80  Inet6VpnRoute vpn_route(vpn_prefix);
81  DBTablePartition *partition =
82  static_cast<DBTablePartition *>(GetTablePartition(&vpn_route));
83  BgpRoute *dest_route = static_cast<BgpRoute *>(partition->Find(&vpn_route));
84  if (dest_route == NULL) {
85  dest_route = new Inet6VpnRoute(vpn_prefix);
86  partition->Add(dest_route);
87  } else {
88  dest_route->ClearDelete();
89  }
90 
91  BgpAttrPtr new_attr =
92  server->attr_db()->ReplaceExtCommunityAndLocate(src_path->GetAttr(),
93  community);
94 
95  // Check whether there's already a path with the given peer and path id.
96  BgpPath *dest_path =
97  dest_route->FindSecondaryPath(source_rt, src_path->GetSource(),
98  src_path->GetPeer(),
99  src_path->GetPathId());
100  if (dest_path != NULL) {
101  if ((new_attr != dest_path->GetOriginalAttr()) ||
102  (src_path->GetFlags() != dest_path->GetFlags()) ||
103  (src_path->GetLabel() != dest_path->GetLabel())) {
104  // Update Attributes and notify (if needed)
105  assert(dest_route->RemoveSecondaryPath(source_rt,
106  src_path->GetSource(), src_path->GetPeer(),
107  src_path->GetPathId()));
108  } else {
109  return dest_route;
110  }
111  }
112 
113  // Create replicated path and insert it on the route
114  BgpSecondaryPath *replicated_path =
115  new BgpSecondaryPath(src_path->GetPeer(), src_path->GetPathId(),
116  src_path->GetSource(), new_attr,
117  src_path->GetFlags(), src_path->GetLabel());
118  replicated_path->SetReplicateInfo(src_table, source_rt);
119  dest_route->InsertPath(replicated_path);
120 
121  // Always trigger notification.
122  partition->Notify(dest_route);
123 
124  return dest_route;
125 }
