156  } else if (result.find("CentOS distro 6.5") != std::string::npos) {
157  distro = "CentOS";

159  } else if (result.find("CentOS Linux release 7") != std::string::npos) {
160  distro = "CentOS";
161  code_name = "7.1.1503";
162  } else {
163  return false;
164  }
165  fclose(fp);
166  return true;
