81 bool TcpServer::InitializeInternal(tcp::endpoint localaddr) {
82  acceptor_.reset(new tcp::acceptor(*evm_->io_service()));
83  if (!acceptor_) {
84  TCP_SERVER_LOG_ERROR(this, TCP_DIR_NA, "Cannot create acceptor");
85  return false;
86  }
87 
88  error_code ec;
89  if (localaddr.address().is_v4())
90  acceptor_->open(tcp::v4(), ec);
91  else
92  acceptor_->open(tcp::v6(), ec);
93 
94  if (ec) {
95  TCP_SERVER_LOG_ERROR(this, TCP_DIR_NA, "TCP open: " << ec.message());
96  ResetAcceptor();
97  return false;
98  }
99 
100  acceptor_->set_option(socket_base::reuse_address(true), ec);
101  if (ec) {
102  TCP_SERVER_LOG_ERROR(this, TCP_DIR_NA, "TCP reuse_address: "
103  << ec.message());
104  ResetAcceptor();
105  return false;
106  }
107 
108  acceptor_->bind(localaddr, ec);
109  if (ec) {
110  TCP_SERVER_LOG_ERROR(this, TCP_DIR_NA, "TCP bind(" << localaddr.address() <<
111  ":" << localaddr.port() << "): " << ec.message());
112  ResetAcceptor();
113  return false;
114  }
115 
116  tcp::endpoint local_endpoint = acceptor_->local_endpoint(ec);
117  if (ec) {
118  TCP_SERVER_LOG_ERROR(this, TCP_DIR_NA,
119  "Cannot retrieve acceptor local-endpont");
120  ResetAcceptor();
121  return false;
122  }
123 
124  //
125  // Server name can be set after local-endpoint information is available.
126  //
127  SetName(local_endpoint);
128 
129  acceptor_->listen(socket_base::max_connections, ec);
130  if (ec) {
131  TCP_SERVER_LOG_ERROR(this, TCP_DIR_NA, "TCP listen(" << localaddr.port() <<
132  "): " << ec.message());
133  ResetAcceptor();
134  return false;
135  }
136 
137  TCP_SERVER_LOG_DEBUG(this, TCP_DIR_NA, "Initialization complete");
138  AsyncAccept();
139 
140  return true;
141 }

349 //
350 // concurrency: called from the event_manager thread.
351 //
352 // accept() tcp connections. Once done, must register with boost again
353 // via AsyncAccept() in order to process future accept calls
354 //
355 void TcpServer::AcceptHandlerInternal(TcpServerPtr server,
356  const error_code& error) {
357  tcp::endpoint remote;
358  error_code ec;
359  TcpSessionPtr session;
360  bool need_close = false;
361 
362  if (error) {
363  goto done;
364  }
365 
366  remote = accept_socket()->remote_endpoint(ec);
367  if (ec) {
368  TCP_SERVER_LOG_ERROR(this, TCP_DIR_IN,
369  "Accept: No remote endpoint: " << ec.message());
370  goto done;
371  }
372 
373  if (acceptor_ == NULL) {
374  TCP_SESSION_LOG_DEBUG(session, TCP_DIR_IN,
375  "Session accepted after server shutdown: "
376  << remote.address().to_string()
377  << ":" << remote.port());
378  accept_socket()->close(ec);
379  goto done;
380  }
381 
382  session.reset(AllocSession(true));
383  if (session == NULL) {
384  TCP_SERVER_LOG_DEBUG(this, TCP_DIR_IN, "Session not created");
385  goto done;
386  }
387 
388  ec = session->SetSocketOptions();
389  if (ec) {
390  TCP_SESSION_LOG_ERROR(session, TCP_DIR_IN,
391  "Accept: Non-blocking error: " << ec.message());
392  need_close = true;
393  goto done;
394  }
395 
396  session->SessionEstablished(remote, TcpSession::PASSIVE);
397  AcceptHandlerComplete(session);
398 
399 done:
400  if (need_close) {
401  session->CloseInternal(ec, false, false);
402  }
403  AsyncAccept();
404 }

406 void TcpServer::AcceptHandlerComplete(TcpSessionPtr session) {
407  tcp::endpoint remote = session->remote_endpoint();
408  {
409  std::scoped_lock lock(mutex_);
410  if (AcceptSession(session.get())) {
411  TCP_SESSION_LOG_UT_DEBUG(session, TCP_DIR_IN,
412  "Accepted session from "
413  << remote.address().to_string()
414  << ":" << remote.port());
415  session_ref_.insert(session);
416  InsertSessionToMap(remote, session.get());
417  } else {
418  TCP_SESSION_LOG_UT_DEBUG(session, TCP_DIR_IN,
419  "Rejected session from "
420  << remote.address().to_string()
421  << ":" << remote.port());
422  error_code ec;
423  session->CloseInternal(ec, false, false);
424  return;
425  }
426  }
427 
428  session->Accepted();
429 }

453  error_code ec;
454  Endpoint remote = session->socket()->remote_endpoint(ec);

483 int TcpServer::SetMd5SocketOption(NativeSocketType fd, uint32_t peer_ip,
484  const string &md5_password) {
485  assert(md5_password.size() <= TCP_MD5SIG_MAXKEYLEN);
486  if (!peer_ip) {
487  TCP_SERVER_LOG_ERROR(this, TCP_DIR_NA, "Invalid peer IP");
488  return 0;
489  }
490 
491  struct sockaddr_in local_addr;
492  memset(&local_addr, 0, sizeof(local_addr));
493 
494  local_addr.sin_family = AF_INET;
495  local_addr.sin_addr.s_addr = htonl(peer_ip);
496 
497  struct tcp_md5sig md5sig;
498  memset(&md5sig, 0, sizeof (md5sig));
499 
500  memcpy(md5sig.tcpm_key, md5_password.c_str(), md5_password.size());
501  md5sig.tcpm_keylen = md5_password.size();
502  memcpy(&md5sig.tcpm_addr, &local_addr, sizeof(local_addr));
503  int retval = setsockopt(fd, IPPROTO_TCP, TCP_MD5SIG, (const char *)&md5sig,
504  sizeof(md5sig));
505  if (retval < 0) {
506  TCP_SERVER_LOG_ERROR(this, TCP_DIR_NA,
507  "Failure in setting md5 key on the socket " +
508  integerToString(fd) + " for peer " + integerToString(peer_ip) +
509  " with errno " + strerror(errno));
510  } else {
511  TCP_SERVER_LOG_DEBUG(this, TCP_DIR_NA,
512  "Success in setting md5 key on the socket " +
513  integerToString(fd) + " for peer " + integerToString(peer_ip));
514  }
515  return retval;
516 }

536 int TcpServer::SetDscpSocketOption(NativeSocketType fd, uint8_t value) {
537  /* The 'value' argument is expected to have DSCP value between 0 and 63 ie
538  * in the lower order 6 bits of a byte. However, setsockopt expects DSCP
539  * value in upper 6 bits of a byte. Hence left shift the value by 2 digits
540  * before passing it to setsockopt */
541  value = value << 2;
542  int retval = setsockopt(fd, IPPROTO_IP, IP_TOS,
543  reinterpret_cast<const char *>(&value), sizeof(value));
544  if (retval < 0) {
545  TCP_SERVER_LOG_ERROR(this, TCP_DIR_NA,
546  "Failure in setting DSCP value on the socket " +
547  integerToString(fd) + " for value " + integerToString(value) +
548  " with errno " + strerror(errno));
549  }
550  return retval;
551 }
