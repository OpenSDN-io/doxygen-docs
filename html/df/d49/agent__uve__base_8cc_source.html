51 void AgentUveBase::BuildTagNamesFromList(const TagList &tl, UveTagData *info)
52  const {
53  TagTable *table = agent_->tag_table();
54  TagList::const_iterator it = tl.begin();
55  while (it != tl.end()) {
56  uint32_t type = ((uint32_t)*it >> TagEntry::kTagTypeBitShift);
57  switch (type) {
58  case TagTable::APPLICATION:
59  info->application = table->TagName(*it);
60  break;
61  case TagTable::TIER:
62  info->tier = table->TagName(*it);
63  break;
64  case TagTable::SITE:
65  info->site = table->TagName(*it);
66  break;
67  case TagTable::DEPLOYMENT:
68  info->deployment = table->TagName(*it);
69  break;
70  case TagTable::LABEL:
71  if (info->fill_type == UveTagData::SET) {
72  info->label_set.insert(table->TagName(*it));
73  break;
74  } else if (info->fill_type == UveTagData::VECTOR) {
75  info->label_vector.push_back(table->TagName(*it));
76  break;
77  }
78  if (!info->labels.empty()) {
79  info->labels += ";";
80  }
81  info->labels.append(table->TagName(*it));
82  break;
83  case TagTable::NEUTRON_FWAAS:
84  info->application = table->TagName(*it);
85  break;
86  default:
87  if (info->fill_type == UveTagData::SET) {
88  info->custom_tag_set.insert(table->TagName(*it));
89  break;
90  } else if (info->fill_type == UveTagData::VECTOR) {
91  info->custom_tag_vector.push_back(table->TagName(*it));
92  break;
93  }
94  if (!info->custom_tags.empty()) {
95  info->custom_tags += ";";
96  }
97  info->custom_tags.append(table->TagName(*it));
98  break;
99  }
100  ++it;
101  }
102 }

237 void AgentUveBase::VrouterAgentProcessState
238  (const std::vector<ConnectionInfo> &cinfos,
239  ProcessState::type &pstate, std::string &message) {
240  size_t num_conns = 0;
241  uint8_t num_control_nodes = 0, num_dns_servers = 0;
242  uint8_t down_control_nodes = 0;
243  uint8_t expected_conns = ExpectedConnections(num_control_nodes,
244  num_dns_servers);
245  std::string cup(g_process_info_constants.ConnectionStatusNames.
246  find(ConnectionStatus::UP)->second);
247  bool is_cup = true;
248  bool is_tor_connected = false;
249  string tor_type(g_process_info_constants.ConnectionTypeNames.
250  find(ConnectionType::TOR)->second);
251  // Iterate to determine process connectivity status
252  for (std::vector<ConnectionInfo>::const_iterator it = cinfos.begin();
253  it != cinfos.end(); it++) {
254  const ConnectionInfo &cinfo(*it);
255  const std::string &conn_status(cinfo.get_status());
256  if (cinfo.get_type() == tor_type) {
257  is_tor_connected = true;
258  continue;
259  }
260  /* Don't consider ConnectionType::TOR type for counting connections.
261  * contrail-tor-agent is not supposed to report as Non-Functional when
262  * it is in backup mode, but contrail-tor-agent does not have a way to
263  * figure out that it is in backup mode. Hence for contrail-tor-agent
264  * (both active and backup modes) we don't consider connection to TOR
265  * for reporting Node Status */
266  num_conns++;
267  if (conn_status != cup) {
268  if (cinfo.get_name().compare(0, 13,
269  agent_->xmpp_control_node_prefix()) == 0) {
270  down_control_nodes++;
271  }
272  is_cup = false;
273  UpdateMessage(cinfo, message);
274  }
275  }
276  if ((num_control_nodes == 0) || (num_control_nodes == down_control_nodes)) {
277  pstate = ProcessState::NON_FUNCTIONAL;
278  if ((num_control_nodes == 0) && message.empty()) {
279  message = "No control-nodes configured";
280  }
281  } else if (!is_tor_connected && agent_->tor_agent_enabled()) {
282  // waiting for first TOR config to arrive
283  pstate = ProcessState::NON_FUNCTIONAL;
284  message += " No ToR Config";
285  } else {
286  pstate = ProcessState::FUNCTIONAL;
287  }
288  if (!is_cup) {
289  message += " connection down";
290  }
291  if (!HasSelfConfiguration()) {
292  // waiting for Global vrouter config
293  pstate = ProcessState::NON_FUNCTIONAL;
294  if (message.empty()) {
295  message = "No Configuration for self";
296  } else {
297  message += ", No Configuration for self";
298  }
299  }
300  for (int i = 0; i < MAX_XMPP_SERVERS; i++) {
301  if (!agent_->controller_ifmap_xmpp_server(i).empty()) {
302  if (agent_->stats()->xmpp_reconnects(i) >= 1) {
303  break;
304  }
305  }
306  }
307 
308  if (num_conns != expected_conns) {
309  message += " Number of connections:" + integerToString(num_conns) +
310  ", Expected: " + integerToString(expected_conns);
311  return;
312  }
313  return;
314 }
