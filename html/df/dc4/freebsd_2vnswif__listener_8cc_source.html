55 unsigned int
56 VnswInterfaceListenerFreeBSD::RTMGetAddresses(const char *in, size_t *size,
57  unsigned int af, struct rt_addresses *rta) {
58 
59  struct sockaddr **out = (struct sockaddr **)rta;
60  int i = 0;
61  size_t wsize = *size;
62  unsigned int oaf = af;
63 
64  if (!(af & ((unsigned int)-1 >> (32 - RTM_ADDR_MAX))))
65  return (af & (unsigned int)-1 << RTM_ADDR_MAX);
66 
67  while (i < (int)RTM_ADDR_MAX && af != 0 && wsize != 0) {
68  if (wsize < SA_SIZE(in)) {
69  break;
70  }
71 
72  if (af & 1) {
73  *out = (struct sockaddr *)in;
74  wsize -= SA_SIZE(in);
75  in += SA_SIZE(in);
76  }
77  i++;
78  out++;
79  af >>= 1;
80  }
81  oaf &= (unsigned int)-1 << i;
82 
83  *size -= wsize;
84 
85  return oaf;
86 }

120 VnswInterfaceListener::Event *
121 VnswInterfaceListenerFreeBSD::RTMProcess(const struct rt_msghdr *rtm,
122  size_t size) {
123 
124  struct rt_addresses rta = { 0 };
125  size_t s = size - sizeof(*rtm);
126  char name[IFNAMSIZ] = { 0 };
127  Event::Type type;
128  unsigned int mask_len = 0;
129  struct sockaddr_in *t;
130 
131  /* This actually is an error */
132  if (sizeof(*rtm) > size) {
133  LOG(ERROR, __PRETTY_FUNCTION__ << ": message too short");
134  return NULL;
135  }
136 
137  /* Not interested in self generated messages */
138  if (rtm->rtm_pid == pid_)
139  return NULL;
140 
141  /* Special case: RTM_DELETE may come without gateway and interface
142  index. Do not serve that, event requires if name na gateway
143  that will not be provided */
144  if (rtm->rtm_type == RTM_DELETE &&
145  !(rtm->rtm_addrs & RTA_GATEWAY))
146  return NULL;
147 
148  if (RTMGetAddresses((char *)(rtm + 1) , &s,
149  rtm->rtm_addrs, &rta) != 0) {
150  LOG(ERROR, __PRETTY_FUNCTION__
151  << RTMTypeToString(rtm->rtm_type)
152  << "IP address extraction failed.");
153  return NULL;
154  }
155 
156  if (rtm->rtm_addrs & RTA_NETMASK) {
157  mask_len = NetmaskLen(
158  ntohl(((struct sockaddr_in *)rta.netmask)->sin_addr.s_addr)
159  );
160  }
161 
162  /* Only routes to host are supported, which means routes that
163  either have RTF_HOST flags or netmask narrowed to one host
164  (unicast, all 1s) */
165  if (!(rtm->rtm_flags & RTF_HOST || mask_len == 32)) {
166  LOG(DEBUG, __PRETTY_FUNCTION__ << ": "
167  << RTMTypeToString(rtm->rtm_type)
168  << " misses requirements for processing");
169 
170  return NULL;
171  }
172 
173  /* There might be no interface name within address block, but there
174  is always interface index that can be used to get the name */
175  if (rtm->rtm_addrs & RTA_IFP && rta.ifp != NULL) {
176  struct sockaddr_dl *sd = (struct sockaddr_dl *)rta.ifp;
177  strncpy(name, sd->sdl_data, sd->sdl_nlen);
178  } else {
179  if_indextoname(rtm->rtm_index, name);
180  }
181 
182  t = (struct sockaddr_in *)rta.dst;
183  Ip4Address dst_addr((unsigned long)ntohl(t->sin_addr.s_addr));
184  t = (struct sockaddr_in *)rta.gw;
185  Ip4Address gw_addr((unsigned long)ntohl(t->sin_addr.s_addr));
186 
187  if (rtm->rtm_type == RTM_DELETE) {
188  type = Event::DEL_ROUTE;
189  } else {
190  /* Handles also RTM_GET message, which is similar in structure
191  but is used for reporting current state */
192  type = Event::ADD_ROUTE;
193  }
194 
195  /* "Protocol" the route has been set with */
196  char proto = rtm->rtm_flags & (RTF_STATIC | RTF_DYNAMIC);
197 
198  LOG(DEBUG, __PRETTY_FUNCTION__ << ": "
199  << RTMTypeToString(rtm->rtm_type)
200  << " : " << dst_addr.to_string() << "/" << mask_len
201  << " Interface " << name << " GW " << gw_addr.to_string());
202 
203  return new Event(type, dst_addr, mask_len, name, gw_addr,
204  proto, rtm->rtm_flags);
205 }

300 int VnswInterfaceListenerFreeBSD::RTMDecode(const struct rt_msghdr_common *rtm,
301  size_t len, uint32_t seq_no) {
302  Event *event = NULL;
303  /* Segfault protection */
304  if (len < sizeof(*rtm)) {
305  return 1;
306  }
307 
308  switch (rtm->rtmc_type) {
309  case RTM_GET:
310  case RTM_ADD:
311  case RTM_DELETE:
312  event = RTMProcess((const struct rt_msghdr *)rtm, len);
313  break;
314  case RTM_NEWADDR:
315  case RTM_DELADDR:
316  event = RTMProcess((const struct ifa_msghdr *)rtm, len);
317  break;
318  case RTM_IFINFO:
319  event = RTMProcess((const struct if_msghdr *)rtm, len);
320  break;
321  default:
322  LOG(DEBUG, __PRETTY_FUNCTION__ << ": message "
323  << RTMTypeToString(rtm->rtmc_type));
324  break;
325  }
326 
327  if (event != NULL) {
328  revent_queue_->Enqueue(event);
329  }
330  return 0;
331 }

334  size_t size) {
335  struct rt_msghdr_common *p = (struct rt_msghdr_common *)buffer;
336  int ret = 0;
337 
338  while (size != 0 && size >= p->rtmc_msglen && p->rtmc_msglen > 0) {
339  ret = RTMDecode(p, size, 0);
340 
341  if (ret)
342  break;
343 
344  size -= p->rtmc_msglen;
345  p = (struct rt_msghdr_common *)((char *)p + p->rtmc_msglen);
346  }
347 
348  /* This is not an error since size of buffer, instead of size of
349  all messages in that buffer could be given. */
350  if (size > 0)
351  LOG(DEBUG, __PRETTY_FUNCTION__
352  << ": message data left unprocessed(" << size << " bytes)");
353 
354  return ret;

522 void VnswInterfaceListenerFreeBSD::UpdateLinkLocalRoute(
523  const Ip4Address &addr, uint8_t plen, bool del_rt) {
524 
525  if (agent_->test_mode())
526  return;
527 
528  memset(tx_buf_, 0, kMaxBufferSize);
529 
530  size_t size;
531  struct rt_msghdr *rtm = (struct rt_msghdr *)tx_buf_;
532  struct sockaddr_in *si = (struct sockaddr_in *)(rtm + 1);
533  struct sockaddr *gw = NULL;
534  struct ifaddrs *ifap, *oifap;
535  const char *vhost_name = agent_->vhost_interface_name().c_str();
536 
537  /* Need to get the gateway IP, the interface is not enough */
538  if (getifaddrs(&ifap) == -1) {
539  /* We will not be able to set route if we are not able to
540  set gateway. That's some major failure */
541  LOG(ERROR, __PRETTY_FUNCTION__
542  << ": failed to get list of interfaces");
543  assert(0);
544  }
545 
546  oifap = ifap;
547 
548  while (ifap != NULL) {
549  if (ifap->ifa_name != NULL &&
550  !strncmp(ifap->ifa_name, vhost_name, IFNAMSIZ) &&
551  SA_SIZE(ifap->ifa_addr) >= SA_SIZE(NULL) &&
552  ifap->ifa_addr->sa_family == AF_INET)
553  {
554  gw = ifap->ifa_addr;
555  }
556 
557  ifap = ifap->ifa_next;
558  }
559 
560  if (gw == NULL) {
561  /* We will not be able to set route if we are not able to
562  set gateway. That's some major failure */
563  LOG(ERROR, __PRETTY_FUNCTION__ << ": failed to get IP for "
564  << string(vhost_name));
565  assert(gw != NULL);
566  }
567 
568 
569  /* Destination, Gateway and header */
570  size = 2 * sizeof(struct sockaddr_in) + sizeof(struct rt_msghdr);
571 
572  if (size > kMaxBufferSize) {
573  LOG(ERROR, __PRETTY_FUNCTION__
574  << ": tx_buf_ to short, expected size " << size);
575  assert(size < kMaxBufferSize);
576  }
577 
578  rtm->rtm_msglen = size;
579  rtm->rtm_version = RTM_VERSION;
580 
581  if (!del_rt)
582  rtm->rtm_type = RTM_ADD;
583  else
584  rtm->rtm_type = RTM_DELETE;
585 
586  rtm->rtm_flags = RTF_UP|RTF_GATEWAY|RTF_STATIC|RTF_HOST;
587 
588  rtm->rtm_index = if_nametoindex(vhost_name);
589 
590  if (rtm->rtm_index == 0) {
591  LOG(ERROR, __PRETTY_FUNCTION__ << ": Failed to obtain index "
592  << "for interface " << vhost_name);
593  assert(0);
594  }
595 
596  rtm->rtm_addrs = RTA_DST|RTA_GATEWAY;
597  rtm->rtm_pid = pid_;
598  rtm->rtm_seq = seqno_++;
599 
600  /* Destination */
601  si->sin_len = sizeof(struct sockaddr_in);
602  si->sin_family = AF_INET;
603  si->sin_port = 0;
604  si->sin_addr.s_addr = htonl(addr.to_ulong());
605 
606  /* Gateway */
607  si = (struct sockaddr_in *)((char *)si + sizeof(struct sockaddr_in));
608  /* Very unlikely... but hard to detect if happens */
609  if (sizeof(struct sockaddr_in) < SA_SIZE(gw)) {
610  LOG(ERROR, __PRETTY_FUNCTION__
611  << ": Size of gateway address returned by kernel is "
612  << SA_SIZE(gw) << "which far too big for socaddr_in struct");
613  assert(sizeof(struct sockaddr_in) >= SA_SIZE(gw));
614  }
615  memcpy(si, gw, SA_SIZE(gw));
616 
617  freeifaddrs(oifap);
618 
619  boost::system::error_code ec;
620  sock_.send(boost::asio::buffer(tx_buf_, size), 0, ec);
621  if (ec.value() != 0) {
622  LOG(ERROR, __PRETTY_FUNCTION__
623  << ": PF_ROUTE message " << RTMTypeToString(rtm->rtm_type)
624  << " send failed");
625  assert(ec.value() == 0);
626  }
627 }
