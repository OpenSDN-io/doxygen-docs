321  trigger_->Set();

502 // The IFMap dependency tracker accepts rules in the form or "Reactor Map"
503 // Reactor map is not natural way of defining references between IFMap nodes
504 // in agent.
505 //
506 // When agent builds data for an oper-db entry, the IFNodeToReq API will keep
507 // the IFMapNode for oper-db entry as starting node and traverses the graph
508 // to build all realvent data for the entry. So, it is more natural in agent
509 // to specify relation between nodes in a way IFMapNodeToReq traverses the
510 // graph
511 //
512 // The routine AddDependencyPath takes the graph path between two vertices
513 // and then creates "Reactor Map" for them.
514 //
515 // The API takes IFMapNode for object being built as "node" and then a graph
516 // path to reach another IFMapNode of interest.
517 //
518 // Example: VMI refers to RI with following path to support floating-ip
519 // VMI <----> FIP <----> FIP-POOL <----> VN <----> RI
520 //
521 // node is specified as "virtual-machine-interface"
522 // path is specified as,
523 // <string(link-metadata), string(vertex), bool(vertex-attr-interest)>
524 //
525 // vertex specifies name of the adjacent node in the path
526 // link-metadata specifies metadata of the link connecting the vertex
527 // vertex-attr-interest
528 // If "node" is interested in any attribute of vertex, this value
529 // is set to TRUE else its set to FALSE
530 // When value is set to TRUE, it will generate an event of type
531 // "self" in the reactor-map
532 //
533 // The path from virtual-machine-interface to "routing-instance" above
534 // is given as below,
535 //
536 // <"virtual-machine-interface-floating-ip", "floating-ip", true>
537 // <"floating-ip-floating-ip-pool", "floating-ip-pool", false>
538 // <"floating-ip-pool-virtual-network", "virtual-network", true>
539 // <"virtual-network-routing-instance", "routing-instance", true>
540 //
541 // AddDependencyPath will "APPEND" following Reactor rules,
542 // node(virtual-machine-interface)
543 // event(virtual-machine-floating-ip) => Rectors(self)
544 //
545 // node(routing-instance)
546 // event(self) => Reactors(virtual-network-routing-instance)
547 //
548 // node(virtual-network)
549 // event(virtual-network-routing-instance) => Reactors(floating-ip-pool-virtual-network)
550 //
551 // node(floating-ip-pool)
552 // event(self) => Reactors(virtual-machine-interface-virtual-network)
553 // event(floating-ip-pool-virtual-network) => Reactors(floating-ip-floating-ip-pool)
554 //
555 // node(floating-ip)
556 // event(self) => Reactors(virtual-machine-interface-floating-ip)
557 // event(floating-ip-floating-ip-pool) => Reactors(virtual-machine-interface-floating-ip)
558 //
560 void IFMapDependencyManager::AddDependencyPath(const std::string &node,
561  Path path) {
562  NodeEventPolicy *policy = tracker_->policy_map();
563  assert(policy);
564 
565  NodeEventPolicy::iterator node_it;
566  node_it = LocateNodeEventPolicy(policy, node);
567  ReactionMap::iterator react_it = LocateReactionMap(&node_it->second,
568  path[0].edge_);
569  LOG(DEBUG, "Updating dependency tacker rules for " << node);
570  react_it->second.insert("self");
571  LOG(DEBUG, "Adding ReactorMap " << node_it->first << " : " <<
572  react_it->first << " -> " << "self");
573 
574  for (size_t i = 0; i < path.size(); i++) {
575  node_it = LocateNodeEventPolicy(policy, path[i].vertex_);
576  if (path[i].vertex_interest_) {
577  react_it = LocateReactionMap(&node_it->second, "self");
578  react_it->second.insert(path[i].edge_);
579  LOG(DEBUG, "Adding ReactorMap " << node_it->first << " : " <<
580  react_it->first << " -> " << path[i].edge_);
581  }
582 
583  if (i < (path.size() - 1)) {
584  react_it = LocateReactionMap(&node_it->second, path[i+1].edge_);
585  react_it->second.insert(path[i].edge_);
586  LOG(DEBUG, "Adding ReactorMap " << node_it->first << " : " <<
587  react_it->first << " -> " << path[i].edge_);
588  } else {
589  react_it = LocateReactionMap(&node_it->second, path[i].edge_);
590  react_it->second.insert("nil");
591  LOG(DEBUG, "Adding ReactorMap " << node_it->first << " : " <<
592  react_it->first << " -> " << "nil");
593  }
594  }
595 
596  return;
597 }

1073 void IFMapNodePolicyReq::HandleRequest() const {
1074  Agent *agent = Agent::GetInstance();
1075  IFMapNodePolicyResp *resp = new IFMapNodePolicyResp();
1076 
1077  IFMapDependencyTracker::NodeEventPolicy *policy_list =
1078  agent->oper_db()->dependency_manager()->tracker()->policy_map();
1079  NodeEventPolicy::iterator it1 = policy_list->begin();
1080  std::vector<IFMapNodePolicy> resp_policy_list;
1081  while (it1 != policy_list->end()) {
1082  if (it1->first.find(get_node()) == string::npos) {
1083  it1++;
1084  continue;
1085  }
1086 
1087  ReactionMap::iterator it2 = it1->second.begin();
1088  vector<IFMapReactEvent> resp_event_list;
1089  while(it2 != it1->second.end()) {
1090  IFMapReactEvent resp_event;
1091  vector<string> resp_reactor_list;
1092  PropagateList::iterator it3 = it2->second.begin();
1093  while (it3 != it2->second.end()) {
1094  resp_reactor_list.push_back((*it3));
1095  it3++;
1096  }
1097  resp_event.set_event(it2->first);
1098  resp_event.set_reactors(resp_reactor_list);
1099  resp_event_list.push_back(resp_event);
1100  it2++;
1101  }
1102 
1103  IFMapNodePolicy policy;
1104  policy.set_node(it1->first);
1105  policy.set_events(resp_event_list);
1106  resp_policy_list.push_back(policy);
1107  it1++;
1108  }
1109 
1110  resp->set_policies(resp_policy_list);
1111  resp->set_context(context());
1112  resp->set_more(false);
1113  resp->Response();
