113 bool BgpConditionListener::PurgeTableState() {
114  CHECK_CONCURRENCY("bgp::Config");
115  for (auto *ts : purge_list_) {
116  if (ts->match_objects()->empty()) {
117  BgpTable *bgptable = ts->table();
118  if (ts->walk_ref() != NULL)
119  bgptable->ReleaseWalker(ts->walk_ref());
120  bgptable->Unregister(ts->GetListenerId());
121  map_.erase(bgptable);
122  delete ts;
123  }
124  }
125  purge_list_.clear();
126 
127  return true;
128 }

131 // AddMatchCondition:
132 // API to add ConditionMatch object against a table
133 // All entries present in this table will be matched for this ConditionMatch

135 // Match is done either in TableWalk or During Table entry notification

137 void BgpConditionListener::AddMatchCondition(BgpTable *table,
138  ConditionMatch *obj,
139  RequestDoneCb cb) {
140  CHECK_CONCURRENCY("bgp::Config", "bgp::ConfigHelper");
141 
142  tbb::mutex::scoped_lock lock(mutex_);
143  ConditionMatchTableState *ts = NULL;
144  TableMap::iterator loc = map_.find(table);
145  if (loc == map_.end()) {
146  DBTableBase::ListenerId id =
147  table->Register(
148  boost::bind(&BgpConditionListener::BgpRouteNotify,
149  this, server(), _1, _2),
150  "BgpConditionListener");
151  ts = new ConditionMatchTableState(table, id);
152  map_.insert(make_pair(table, ts));
153  } else {
154  ts = loc->second;
155  }
156  ts->AddMatchObject(obj);
157  TableWalk(ts, obj, cb);
158 }

296 void BgpConditionListener::TableWalk(ConditionMatchTableState *ts,
297  ConditionMatch *obj, BgpConditionListener::RequestDoneCb cb) {
298  CHECK_CONCURRENCY("bgp::Config", "bgp::ConfigHelper");
299 
300  if (ts->walk_ref() == NULL) {
301  DBTable::DBTableWalkRef walk_ref = ts->table()->AllocWalker(
302  boost::bind(&BgpConditionListener::BgpRouteNotify,
303  this, server(), _1, _2),
304  boost::bind(&BgpConditionListener::WalkDone,
305  this, ts, _2));
306  ts->set_walk_ref(walk_ref);
307  }
308  ts->StoreDoneCb(obj, cb);
309  obj->reset_walk_done();
310  ts->table()->WalkTable(ts->walk_ref());
311 }

314 bool BgpConditionListener::BgpRouteNotify(BgpServer *server,
315  DBTablePartBase *root,
316  DBEntryBase *entry) {
317  BgpTable *bgptable = static_cast<BgpTable *>(root->parent());
318  BgpRoute *rt = static_cast<BgpRoute *> (entry);
319  // Either the route is deleted or no valid path exists
320  bool del_rt = !rt->IsUsable();
321 
322  TableMap::iterator loc = map_.find(bgptable);
323  assert(loc != map_.end());
324  ConditionMatchTableState *ts = loc->second;
325 
326  DBTableBase::ListenerId id = ts->GetListenerId();
327  assert(id != DBTableBase::kInvalidId);
328 
329  for (ConditionMatchTableState::MatchList::iterator match_obj_it =
330  ts->match_objects()->begin();
331  match_obj_it != ts->match_objects()->end(); match_obj_it++) {
332  bool deleted = false;
333  if ((*match_obj_it)->deleted() || del_rt) {
334  deleted = true;
335  }
336  (*match_obj_it)->Match(server, bgptable, rt, deleted);
337  }
338  return true;
339 }

341 //
342 // WalkComplete function
343 // WalkComplete is invoked only after all walk requests for BgpConditionListener
344 // is served. (say after multiple walkagain, only one WalkDone is invoked)
345 // Invoke the RequestDoneCb for all objects for which walk was started
346 // Clear the walk_list_
347 //
348 void BgpConditionListener::WalkDone(ConditionMatchTableState *ts,
349  DBTableBase *table) {
350  BgpTable *bgptable = static_cast<BgpTable *>(table);
351 
352  // Invoke the RequestDoneCb after the TableWalk
353  for (ConditionMatchTableState::WalkList::iterator walk_it =
354  ts->walk_list()->begin();
355  walk_it != ts->walk_list()->end(); ++walk_it) {
356  walk_it->first->set_walk_done();
357  // If application has registered a WalkDone callback, invoke it
358  if (!walk_it->second.empty()) {
359  walk_it->second(bgptable, walk_it->first.get());
360  }
361  }
362 
363  ts->walk_list()->clear();
364 }
